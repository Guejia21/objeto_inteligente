22def build_environ(scope: Scope, body: bytes) -> dict[str, Any]:
23 """
24 Builds a scope and request body into a WSGI environ object.
25 """
26
27 script_name = scope.get("root_path", "").encode("utf8").decode("latin1")
28 path_info = scope["path"].encode("utf8").decode("latin1")
29 if path_info.startswith(script_name):
30 path_info = path_info[len(script_name) :]
31
32 environ = {
33 "REQUEST_METHOD": scope["method"],
34 "SCRIPT_NAME": script_name,
35 "PATH_INFO": path_info,
36 "QUERY_STRING": scope["query_string"].decode("ascii"),
37 "SERVER_PROTOCOL": f"HTTP/{scope['http_version']}",
38 "wsgi.version": (1, 0),
39 "wsgi.url_scheme": scope.get("scheme", "http"),
40 "wsgi.input": io.BytesIO(body),
41 "wsgi.errors": sys.stdout,
42 "wsgi.multithread": True,
43 "wsgi.multiprocess": True,
44 "wsgi.run_once": False,
45 }
46
47 # Get server name and port - required in WSGI, not in ASGI
48 server = scope.get("server") or ("localhost", 80)
49 environ["SERVER_NAME"] = server[0]
50 environ["SERVER_PORT"] = server[1]
51
52 # Get client IP address
53 if scope.get("client"):
54 environ["REMOTE_ADDR"] = scope["client"][0]
55
56 # Go through headers and make them into environ entries
57 for name, value in scope.get("headers", []):
58 name = name.decode("latin1")
59 if name == "content-length":
60 corrected_name = "CONTENT_LENGTH"
61 elif name == "content-type":
62 corrected_name = "CONTENT_TYPE"
63 else:
64 corrected_name = f"HTTP_{name}".upper().replace("-", "_")
65 # HTTPbis say only ASCII chars are allowed in headers, but we latin1 just in
66 # case
67 value = value.decode("latin1")
68 if corrected_name in environ:
69 value = environ[corrected_name] + "," + value
70 environ[corrected_name] = value
71 return environ
