166 def __call__(self, settings: BaseSettings) -> Dict[str, Any]: # noqa C901
167 """
168 Build environment variables suitable for passing to the Model.
169 """
170 d: Dict[str, Any] = {}
171
172 if settings.__config__.case_sensitive:
173 env_vars: Mapping[str, Optional[str]] = os.environ
174 else:
175 env_vars = {k.lower(): v for k, v in os.environ.items()}
176
177 dotenv_vars = self._read_env_files(settings.__config__.case_sensitive)
178 if dotenv_vars:
179 env_vars = {**dotenv_vars, **env_vars}
180
181 for field in settings.__fields__.values():
182 env_val: Optional[str] = None
183 for env_name in field.field_info.extra['env_names']:
184 env_val = env_vars.get(env_name)
185 if env_val is not None:
186 break
187
188 is_complex, allow_parse_failure = self.field_is_complex(field)
189 if is_complex:
190 if env_val is None:
191 # field is complex but no value found so far, try explode_env_vars
192 env_val_built = self.explode_env_vars(field, env_vars)
193 if env_val_built:
194 d[field.alias] = env_val_built
195 else:
196 # field is complex and there's a value, decode that as JSON, then add explode_env_vars
197 try:
198 env_val = settings.__config__.parse_env_var(field.name, env_val)
199 except ValueError as e:
200 if not allow_parse_failure:
201 raise SettingsError(f'error parsing env var "{env_name}"') from e
202
203 if isinstance(env_val, dict):
204 d[field.alias] = deep_update(env_val, self.explode_env_vars(field, env_vars))
205 else:
206 d[field.alias] = env_val
207 elif env_val is not None:
208 # simplest case, field is not complex, we only need to add the value if it was found
209 d[field.alias] = env_val
210
211 return d

231 def field_is_complex(self, field: ModelField) -> Tuple[bool, bool]:
232 """
233 Find out if a field is complex, and if so whether JSON errors should be ignored
234 """
235 if lenient_issubclass(field.annotation, JsonWrapper):
236 return False, False
237
238 if field.is_complex():
239 allow_parse_failure = False
240 elif is_union(get_origin(field.type_)) and field.sub_fields and any(f.is_complex() for f in field.sub_fields):
241 allow_parse_failure = True
242 else:
243 return False, False
244
245 return True, allow_parse_failure

247 def explode_env_vars(self, field: ModelField, env_vars: Mapping[str, Optional[str]]) -> Dict[str, Any]:
248 """
249 Process env_vars and extract the values of keys containing env_nested_delimiter into nested dictionaries.
250
251 This is applied to a single field, hence filtering by env_var prefix.
252 """
253 prefixes = [f'{env_name}{self.env_nested_delimiter}' for env_name in field.field_info.extra['env_names']]
254 result: Dict[str, Any] = {}
255 for env_name, env_val in env_vars.items():
256 if not any(env_name.startswith(prefix) for prefix in prefixes):
257 continue
258 # we remove the prefix before splitting in case the prefix has characters in common with the delimiter
259 env_name_without_prefix = env_name[self.env_prefix_len :]
260 _, *keys, last_key = env_name_without_prefix.split(self.env_nested_delimiter)
261 env_var = result
262 for key in keys:
263 env_var = env_var.setdefault(key, {})
264 env_var[last_key] = env_val
265
266 return result

281 def __call__(self, settings: BaseSettings) -> Dict[str, Any]:
282 """
283 Build fields from "secrets" files.
284 """
285 secrets: Dict[str, Optional[str]] = {}
286
287 if self.secrets_dir is None:
288 return secrets
289
290 secrets_path = Path(self.secrets_dir).expanduser()
291
292 if not secrets_path.exists():
293 warnings.warn(f'directory "{secrets_path}" does not exist')
294 return secrets
295
296 if not secrets_path.is_dir():
297 raise SettingsError(f'secrets_dir must reference a directory, not a {path_type(secrets_path)}')
298
299 for field in settings.__fields__.values():
300 for env_name in field.field_info.extra['env_names']:
301 path = find_case_path(secrets_path, env_name, settings.__config__.case_sensitive)
302 if not path:
303 # path does not exist, we currently don't return a warning for this
304 continue
305
306 if path.is_file():
307 secret_value = path.read_text().strip()
308 if field.is_complex():
309 try:
310 secret_value = settings.__config__.parse_env_var(field.name, secret_value)
311 except ValueError as e:
312 raise SettingsError(f'error parsing env var "{env_name}"') from e
313
314 secrets[field.alias] = secret_value
315 else:
316 warnings.warn(
317 f'attempted to load secret file "{path}" but found a {path_type(path)} instead.',
318 stacklevel=4,
319 )
320 return secrets

341def find_case_path(dir_path: Path, file_name: str, case_sensitive: bool) -> Optional[Path]:
342 """
343 Find a file within path's directory matching filename, optionally ignoring case.
344 """
345 for f in dir_path.iterdir():
346 if f.name == file_name:
347 return f
348 elif not case_sensitive and f.name.lower() == file_name.lower():
349 return f
