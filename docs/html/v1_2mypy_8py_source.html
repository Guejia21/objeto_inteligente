171 def _pydantic_field_callback(self, ctx: FunctionContext) -> 'Type':
172 """
173 Extract the type of the `default` argument from the Field function, and use it as the return type.
174
175 In particular:
176 * Check whether the default and default_factory argument is specified.
177 * Output an error if both are specified.
178 * Retrieve the type of the argument which is specified, and use it as return type for the function.
179 """
180 default_any_type = ctx.default_return_type
181
182 assert ctx.callee_arg_names[0] == 'default', '"default" is no longer first argument in Field()'
183 assert ctx.callee_arg_names[1] == 'default_factory', '"default_factory" is no longer second argument in Field()'
184 default_args = ctx.args[0]
185 default_factory_args = ctx.args[1]
186
187 if default_args and default_factory_args:
188 error_default_and_default_factory_specified(ctx.api, ctx.context)
189 return default_any_type
190
191 if default_args:
192 default_type = ctx.arg_types[0][0]
193 default_arg = default_args[0]
194
195 # Fallback to default Any type if the field is required
196 if not isinstance(default_arg, EllipsisExpr):
197 return default_type
198
199 elif default_factory_args:
200 default_factory_type = ctx.arg_types[1][0]
201
202 # Functions which use `ParamSpec` can be overloaded, exposing the callable's types as a parameter
203 # Pydantic calls the default factory without any argument, so we retrieve the first item
204 if isinstance(default_factory_type, Overloaded):
205 if MYPY_VERSION_TUPLE > (0, 910):
206 default_factory_type = default_factory_type.items[0]
207 else:
208 # Mypy0.910 exposes the items of overloaded types in a function
209 default_factory_type = default_factory_type.items()[0] # type: ignore[operator]
210
211 if isinstance(default_factory_type, CallableType):
212 ret_type = get_proper_type(default_factory_type.ret_type)
213 if (
214 isinstance(ret_type, Instance)
215 and ret_type.args
216 and all(isinstance(arg, TypeVarType) for arg in ret_type.args)
217 ):
218 # Looks like the default factory is a type like `list` or `dict`, replace all args with `Any`
219 ret_type = ret_type.copy_modified(args=[default_any_type] * len(ret_type.args))
220 return ret_type
221
222 return default_any_type

301 def transform(self) -> None:
302 """
303 Configures the BaseModel subclass according to the plugin settings.
304
305 In particular:
306 * determines the model config and fields,
307 * adds a fields-aware signature for the initializer and construct methods
308 * freezes the class if allow_mutation = False or frozen = True
309 * stores the fields, config, and if the class is settings in the mypy metadata for access by subclasses
310 """
311 ctx = self._ctx
312 info = ctx.cls.info
313
314 self.adjust_validator_signatures()
315 config = self.collect_config()
316 fields = self.collect_fields(config)
317 is_settings = any(get_fullname(base) == BASESETTINGS_FULLNAME for base in info.mro[:-1])
318 self.add_initializer(fields, config, is_settings)
319 self.add_construct_method(fields)
320 self.set_frozen(fields, frozen=config.allow_mutation is False or config.frozen is True)
321 info.metadata[METADATA_KEY] = {
322 'fields': {field.name: field.serialize() for field in fields},
323 'config': config.set_values_dict(),
324 }

326 def adjust_validator_signatures(self) -> None:
327 """When we decorate a function `f` with `pydantic.validator(...), mypy sees
328 `f` as a regular method taking a `self` instance, even though pydantic
329 internally wraps `f` with `classmethod` if necessary.
330
331 Teach mypy this by marking any function whose outermost decorator is a
332 `validator()` call as a classmethod.
333 """
334 for name, sym in self._ctx.cls.info.names.items():
335 if isinstance(sym.node, Decorator):
336 first_dec = sym.node.original_decorators[0]
337 if (
338 isinstance(first_dec, CallExpr)
339 and isinstance(first_dec.callee, NameExpr)
340 and first_dec.callee.fullname == f'{_NAMESPACE}.class_validators.validator'
341 ):
342 sym.node.func.is_class = True

344 def collect_config(self) -> 'ModelConfigData':
345 """
346 Collects the values of the config attributes that are used by the plugin, accounting for parent classes.
347 """
348 ctx = self._ctx
349 cls = ctx.cls
350 config = ModelConfigData()
351 for stmt in cls.defs.body:
352 if not isinstance(stmt, ClassDef):
353 continue
354 if stmt.name == 'Config':
355 for substmt in stmt.defs.body:
356 if not isinstance(substmt, AssignmentStmt):
357 continue
358 config.update(self.get_config_update(substmt))
359 if (
360 config.has_alias_generator
361 and not config.allow_population_by_field_name
362 and self.plugin_config.warn_required_dynamic_aliases
363 ):
364 error_required_dynamic_aliases(ctx.api, stmt)
365 for info in cls.info.mro[1:]: # 0 is the current class
366 if METADATA_KEY not in info.metadata:
367 continue
368
369 # Each class depends on the set of fields in its ancestors
370 ctx.api.add_plugin_dependency(make_wildcard_trigger(get_fullname(info)))
371 for name, value in info.metadata[METADATA_KEY]['config'].items():
372 config.setdefault(name, value)
373 return config

375 def collect_fields(self, model_config: 'ModelConfigData') -> List['PydanticModelField']:
376 """
377 Collects the fields for the model, accounting for parent classes
378 """
379 # First, collect fields belonging to the current class.
380 ctx = self._ctx
381 cls = self._ctx.cls
382 fields = [] # type: List[PydanticModelField]
383 known_fields = set() # type: Set[str]
384 for stmt in cls.defs.body:
385 if not isinstance(stmt, AssignmentStmt): # `and stmt.new_syntax` to require annotation
386 continue
387
388 lhs = stmt.lvalues[0]
389 if not isinstance(lhs, NameExpr) or not is_valid_field(lhs.name):
390 continue
391
392 if not stmt.new_syntax and self.plugin_config.warn_untyped_fields:
393 error_untyped_fields(ctx.api, stmt)
394
395 # if lhs.name == '__config__': # BaseConfig not well handled; I'm not sure why yet
396 # continue
397
398 sym = cls.info.names.get(lhs.name)
399 if sym is None: # pragma: no cover
400 # This is likely due to a star import (see the dataclasses plugin for a more detailed explanation)
401 # This is the same logic used in the dataclasses plugin
402 continue
403
404 node = sym.node
405 if isinstance(node, PlaceholderNode): # pragma: no cover
406 # See the PlaceholderNode docstring for more detail about how this can occur
407 # Basically, it is an edge case when dealing with complex import logic
408 # This is the same logic used in the dataclasses plugin
409 continue
410 if not isinstance(node, Var): # pragma: no cover
411 # Don't know if this edge case still happens with the `is_valid_field` check above
412 # but better safe than sorry
413 continue
414
415 # x: ClassVar[int] is ignored by dataclasses.
416 if node.is_classvar:
417 continue
418
419 is_required = self.get_is_required(cls, stmt, lhs)
420 alias, has_dynamic_alias = self.get_alias_info(stmt)
421 if (
422 has_dynamic_alias
423 and not model_config.allow_population_by_field_name
424 and self.plugin_config.warn_required_dynamic_aliases
425 ):
426 error_required_dynamic_aliases(ctx.api, stmt)
427 fields.append(
428 PydanticModelField(
429 name=lhs.name,
430 is_required=is_required,
431 alias=alias,
432 has_dynamic_alias=has_dynamic_alias,
433 line=stmt.line,
434 column=stmt.column,
435 )
436 )
437 known_fields.add(lhs.name)
438 all_fields = fields.copy()
439 for info in cls.info.mro[1:]: # 0 is the current class, -2 is BaseModel, -1 is object
440 if METADATA_KEY not in info.metadata:
441 continue
442
443 superclass_fields = []
444 # Each class depends on the set of fields in its ancestors
445 ctx.api.add_plugin_dependency(make_wildcard_trigger(get_fullname(info)))
446
447 for name, data in info.metadata[METADATA_KEY]['fields'].items():
448 if name not in known_fields:
449 field = PydanticModelField.deserialize(info, data)
450 known_fields.add(name)
451 superclass_fields.append(field)
452 else:
453 (field,) = (a for a in all_fields if a.name == name)
454 all_fields.remove(field)
455 superclass_fields.append(field)
456 all_fields = superclass_fields + all_fields
457 return all_fields

503 TypeVarId(-1, namespace=ctx.cls.fullname + '.construct')

507 [],
508 obj_type,

539 def set_frozen(self, fields: List['PydanticModelField'], frozen: bool) -> None:
540 """
541 Marks all fields as properties so that attempts to set them trigger mypy errors.
542
543 This is the same approach used by the attrs and dataclasses plugins.
544 """
545 ctx = self._ctx
546 info = ctx.cls.info
547 for field in fields:
548 sym_node = info.names.get(field.name)

550 var = sym_node.node

553 elif isinstance(var, PlaceholderNode) and not ctx.api.final_iteration:
554 # See https://github.com/pydantic/pydantic/issues/5191 to hit this branch for test coverage

572 def get_config_update(self, substmt: AssignmentStmt) -> Optional['ModelConfigData']:
573 """
574 Determines the config update due to a single statement in the Config class definition.
575
576 Warns if a tracked config attribute is set to a value the plugin doesn't know how to interpret (e.g., an int)
577 """
578 lhs = substmt.lvalues[0]
579 if not (isinstance(lhs, NameExpr) and lhs.name in self.tracked_config_fields):
580 return None
581 if lhs.name == 'extra':
582 if isinstance(substmt.rvalue, StrExpr):
583 forbid_extra = substmt.rvalue.value == 'forbid'
584 elif isinstance(substmt.rvalue, MemberExpr):
585 forbid_extra = substmt.rvalue.name == 'forbid'
586 else:
587 error_invalid_config_value(lhs.name, self._ctx.api, substmt)
588 return None
589 return ModelConfigData(forbid_extra=forbid_extra)
590 if lhs.name == 'alias_generator':
591 has_alias_generator = True
592 if isinstance(substmt.rvalue, NameExpr) and substmt.rvalue.fullname == 'builtins.None':
593 has_alias_generator = False
594 return ModelConfigData(has_alias_generator=has_alias_generator)
595 if isinstance(substmt.rvalue, NameExpr) and substmt.rvalue.fullname in ('builtins.True', 'builtins.False'):
596 return ModelConfigData(**{lhs.name: substmt.rvalue.fullname == 'builtins.True'})
597 error_invalid_config_value(lhs.name, self._ctx.api, substmt)
598 return None

601 def get_is_required(cls: ClassDef, stmt: AssignmentStmt, lhs: NameExpr) -> bool:
602 """
603 Returns a boolean indicating whether the field defined in `stmt` is a required field.
604 """
605 expr = stmt.rvalue
606 if isinstance(expr, TempNode):
607 # TempNode means annotation-only, so only non-required if Optional
608 value_type = get_proper_type(cls.info[lhs.name].type)
609 return not PydanticModelTransformer.type_has_implicit_default(value_type)
610 if isinstance(expr, CallExpr) and isinstance(expr.callee, RefExpr) and expr.callee.fullname == FIELD_FULLNAME:
611 # The "default value" is a call to `Field`; at this point, the field is
612 # only required if default is Ellipsis (i.e., `field_name: Annotation = Field(...)`) or if default_factory
613 # is specified.
614 for arg, name in zip(expr.args, expr.arg_names):
615 # If name is None, then this arg is the default because it is the only positional argument.
616 if name is None or name == 'default':
617 return arg.__class__ is EllipsisExpr
618 if name == 'default_factory':
619 return False

622 return not PydanticModelTransformer.type_has_implicit_default(value_type)

626 @staticmethod
627 def type_has_implicit_default(type_: Optional[ProperType]) -> bool:
628 """
629 Returns True if the passed type will be given an implicit default value.
630
631 In pydantic v1, this is the case for Optional types and Any (with default value None).
632 """
633 if isinstance(type_, AnyType):
634 # Annotated as Any
635 return True
636 if isinstance(type_, UnionType) and any(
637 isinstance(item, NoneType) or isinstance(item, AnyType) for item in type_.items
638 ):
639 # Annotated as Optional, or otherwise having NoneType or AnyType in the union
640 return True

643 @staticmethod
644 def get_alias_info(stmt: AssignmentStmt) -> Tuple[Optional[str], bool]:
645 """
646 Returns a pair (alias, has_dynamic_alias), extracted from the declaration of the field defined in `stmt`.
647
648 `has_dynamic_alias` is True if and only if an alias is provided, but not as a string literal.
649 If `has_dynamic_alias` is True, `alias` will be None.
650 """
651 expr = stmt.rvalue
652 if isinstance(expr, TempNode):
653 # TempNode means annotation-only
654 return None, False
655
656 if not (
657 isinstance(expr, CallExpr) and isinstance(expr.callee, RefExpr) and expr.callee.fullname == FIELD_FULLNAME
658 ):
659 # Assigned value is not a call to pydantic.fields.Field
660 return None, False
661
662 for i, arg_name in enumerate(expr.arg_names):
663 if arg_name != 'alias':
664 continue
665 arg = expr.args[i]
666 if isinstance(arg, StrExpr):
667 return arg.value, False
668 else:
669 return None, True
670 return None, False

824def add_method(
825 ctx: ClassDefContext,
826 name: str,
827 args: List[Argument],
828 return_type: Type,
829 self_type: Optional[Type] = None,
830 tvar_def: Optional[TypeVarDef] = None,
831 is_classmethod: bool = False,
832 is_new: bool = False,
833 # is_staticmethod: bool = False,
834) -> None:
835 """
836 Adds a new method to a class.
837
838 This can be dropped if/when https://github.com/python/mypy/issues/7301 is merged
839 """
840 info = ctx.cls.info
841
842 # First remove any previously generated methods with the same name
843 # to avoid clashes and problems in the semantic analyzer.
844 if name in info.names:
845 sym = info.names[name]
846 if sym.plugin_generated and isinstance(sym.node, FuncDef):
847 ctx.cls.defs.body.remove(sym.node) # pragma: no cover
848
849 self_type = self_type or fill_typevars(info)
850 if is_classmethod or is_new:
851 first = [Argument(Var('_cls'), TypeType.make_normalized(self_type), None, ARG_POS)]
852 # elif is_staticmethod:
853 # first = []
854 else:
855 self_type = self_type or fill_typevars(info)
856 first = [Argument(Var('__pydantic_self__'), self_type, None, ARG_POS)]
857 args = first + args
858 arg_types, arg_names, arg_kinds = [], [], []
859 for arg in args:
860 assert arg.type_annotation, 'All arguments must be fully typed.'
861 arg_types.append(arg.type_annotation)
862 arg_names.append(get_name(arg.variable))
863 arg_kinds.append(arg.kind)
864
865 function_type = ctx.api.named_type(f'{BUILTINS_NAME}.function')
866 signature = CallableType(
867 arg_types, arg_kinds, arg_names, return_type, function_type, variables=[tvar_def] if tvar_def else None
868 )
869
870 func = FuncDef(name, args, Block([PassStmt()]))
871 func.info = info
872 func.type = set_callable_name(signature, func)
873 func.is_class = is_classmethod
874 # func.is_static = is_staticmethod
875 func._fullname = get_fullname(info) + '.' + name
876 func.line = info.line
877
878 # NOTE: we would like the plugin generated node to dominate, but we still
879 # need to keep any existing definitions so they get semantically analyzed.
880 if name in info.names:
881 # Get a nice unique name instead.
882 r_name = get_unique_redefinition_name(name, info.names)
883 info.names[r_name] = info.names[name]
884
885 if is_classmethod: # or is_staticmethod:
886 func.is_decorated = True
887 v = Var(name, func.type)
888 v.info = info
889 v._fullname = func._fullname
890 # if is_classmethod:
891 v.is_classmethod = True
892 dec = Decorator(func, [NameExpr('classmethod')], v)
893 # else:
894 # v.is_staticmethod = True
895 # dec = Decorator(func, [NameExpr('staticmethod')], v)
896
897 dec.line = info.line
898 sym = SymbolTableNode(MDEF, dec)
899 else:
900 sym = SymbolTableNode(MDEF, func)
901 sym.plugin_generated = True
902
903 info.names[name] = sym
904 info.defn.defs.body.append(func)
