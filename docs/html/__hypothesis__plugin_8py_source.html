1"""
2Register Hypothesis strategies for Pydantic custom types.
3
4This enables fully-automatic generation of test data for most Pydantic classes.
5
6Note that this module has *no* runtime impact on Pydantic itself; instead it
7is registered as a setuptools entry point and Hypothesis will import it if
8Pydantic is installed. See also:
9
10https://hypothesis.readthedocs.io/en/latest/strategies.html#registering-strategies-via-setuptools-entry-points
11https://hypothesis.readthedocs.io/en/latest/data.html#hypothesis.strategies.register_type_strategy
12https://hypothesis.readthedocs.io/en/latest/strategies.html#interaction-with-pytest-cov
13https://docs.pydantic.dev/usage/types/#pydantic-types
14
15Note that because our motivation is to *improve user experience*, the strategies
16are always sound (never generate invalid data) but sacrifice completeness for
17maintainability (ie may be unable to generate some tricky but valid data).
18
19Finally, this module makes liberal use of `# type: ignore[<code>]` pragmas.
20This is because Hypothesis annotates `register_type_strategy()` with
21`(T, SearchStrategy[T])`, but in most cases we register e.g. `ConstrainedInt`
22to generate instances of the builtin `int` type which match the constraints.
23"""

58 def is_valid_email(s: str) -> bool:
59 # Hypothesis' st.emails() occasionally generates emails like 0@A0--0.ac
60 # that are invalid according to email-validator, so we filter those out.
61 try:
62 email_validator.validate_email(s, check_deliverability=False)
63 return True
64 except email_validator.EmailNotValidError: # pragma: no cover
65 return False

122def add_luhn_digit(card_number: str) -> str:
123 # See https://en.wikipedia.org/wiki/Luhn_algorithm
124 for digit in '0123456789':
125 with contextlib.suppress(Exception):
126 pydantic.PaymentCardNumber.validate_luhn_check_digit(card_number + digit)
127 return card_number + digit
128 raise AssertionError('Unreachable') # pragma: no cover
129

135 '3[47][0-9]{12}', # American Express
136 '[0-26-9][0-9]{10,17}', # other (incomplete to avoid overlap)

176# Constrained-type resolver functions
177#
178# For these ones, we actually want to inspect the type in order to work out a
179# satisfying strategy. First up, the machinery for tracking resolver functions:

181RESOLVERS: Dict[type, Callable[[type], st.SearchStrategy]] = {} # type: ignore[type-arg]

194def _registered(
195 typ: Union[Type[pydantic.types.T], pydantic.types.ConstrainedNumberMeta]
196) -> Union[Type[pydantic.types.T], pydantic.types.ConstrainedNumberMeta]:
197 # This function replaces the version in `pydantic.types`, in order to
198 # effect the registration of new constrained types so that Hypothesis
199 # can generate valid examples.
200 pydantic.types._DEFINED_TYPES.add(typ)
201 for supertype, resolver in RESOLVERS.items():
202 if issubclass(typ, supertype):
203 st.register_type_strategy(typ, resolver(typ)) # type: ignore
204 return typ
205 raise NotImplementedError(f'Unknown type {typ!r} has no resolver to register') # pragma: no cover
206

208def resolves(
209 typ: Union[type, pydantic.types.ConstrainedNumberMeta]
210) -> Callable[[Callable[..., st.SearchStrategy]], Callable[..., st.SearchStrategy]]: # type: ignore[type-arg]
211 def inner(f): # type: ignore
212 assert f not in RESOLVERS
213 RESOLVERS[typ] = f
214 return f
215
216 return inner

222@resolves(pydantic.JsonWrapper)
223def resolve_json(cls): # type: ignore[no-untyped-def]
224 try:
225 inner = st.none() if cls.inner_type is None else st.from_type(cls.inner_type)
226 except Exception: # pragma: no cover
227 finite = st.floats(allow_infinity=False, allow_nan=False)
228 inner = st.recursive(
229 base=st.one_of(st.none(), st.booleans(), st.integers(), finite, st.text()),
230 extend=lambda x: st.lists(x) | st.dictionaries(st.text(), x), # type: ignore
231 )
232 inner_type = getattr(cls, 'inner_type', None)
233 return st.builds(
234 cls.inner_type.json if lenient_issubclass(inner_type, pydantic.BaseModel) else json.dumps,
235 inner,
236 ensure_ascii=st.booleans(),
237 indent=st.none() | st.integers(0, 16),
238 sort_keys=st.booleans(),
239 )

242@resolves(pydantic.ConstrainedBytes)
243def resolve_conbytes(cls): # type: ignore[no-untyped-def] # pragma: no cover
244 min_size = cls.min_length or 0
245 max_size = cls.max_length
246 if not cls.strip_whitespace:
247 return st.binary(min_size=min_size, max_size=max_size)
248 # Fun with regex to ensure we neither start nor end with whitespace
249 repeats = '{{{},{}}}'.format(
250 min_size - 2 if min_size > 2 else 0,
251 max_size - 2 if (max_size or 0) > 2 else '',
252 )
253 if min_size >= 2:
254 pattern = rf'\W.{repeats}\W'
255 elif min_size == 1:
256 pattern = rf'\W(.{repeats}\W)?'
257 else:
258 assert min_size == 0
259 pattern = rf'(\W(.{repeats}\W)?)?'
260 return st.from_regex(pattern.encode(), fullmatch=True)

263@resolves(pydantic.ConstrainedDecimal)
264def resolve_condecimal(cls): # type: ignore[no-untyped-def]
265 min_value = cls.ge
266 max_value = cls.le
267 if cls.gt is not None:
268 assert min_value is None, 'Set `gt` or `ge`, but not both'
269 min_value = cls.gt
270 if cls.lt is not None:
271 assert max_value is None, 'Set `lt` or `le`, but not both'
272 max_value = cls.lt
273 s = st.decimals(min_value, max_value, allow_nan=False, places=cls.decimal_places)
274 if cls.lt is not None:
275 s = s.filter(lambda d: d < cls.lt)
276 if cls.gt is not None:
277 s = s.filter(lambda d: cls.gt < d)

281@resolves(pydantic.ConstrainedFloat)
282def resolve_confloat(cls): # type: ignore[no-untyped-def]
283 min_value = cls.ge
284 max_value = cls.le
285 exclude_min = False
286 exclude_max = False
287
288 if cls.gt is not None:
289 assert min_value is None, 'Set `gt` or `ge`, but not both'
290 min_value = cls.gt
291 exclude_min = True
292 if cls.lt is not None:
293 assert max_value is None, 'Set `lt` or `le`, but not both'
294 max_value = cls.lt
295 exclude_max = True
296
297 if cls.multiple_of is None:
298 return st.floats(min_value, max_value, exclude_min=exclude_min, exclude_max=exclude_max, allow_nan=False)
299
300 if min_value is not None:
301 min_value = math.ceil(min_value / cls.multiple_of)
302 if exclude_min:
303 min_value = min_value + 1
304 if max_value is not None:
305 assert max_value >= cls.multiple_of, 'Cannot build model with max value smaller than multiple of'
306 max_value = math.floor(max_value / cls.multiple_of)
307 if exclude_max:
308 max_value = max_value - 1
309
310 return st.integers(min_value, max_value).map(lambda x: x * cls.multiple_of)

313@resolves(pydantic.ConstrainedInt)
314def resolve_conint(cls): # type: ignore[no-untyped-def]
315 min_value = cls.ge
316 max_value = cls.le
317 if cls.gt is not None:
318 assert min_value is None, 'Set `gt` or `ge`, but not both'
319 min_value = cls.gt + 1
320 if cls.lt is not None:
321 assert max_value is None, 'Set `lt` or `le`, but not both'
322 max_value = cls.lt - 1
323
324 if cls.multiple_of is None or cls.multiple_of == 1:
325 return st.integers(min_value, max_value)
326
327 # These adjustments and the .map handle integer-valued multiples, while the
328 # .filter handles trickier cases as for confloat.
329 if min_value is not None:
330 min_value = math.ceil(Fraction(min_value) / Fraction(cls.multiple_of))
331 if max_value is not None:
332 max_value = math.floor(Fraction(max_value) / Fraction(cls.multiple_of))
333 return st.integers(min_value, max_value).map(lambda x: x * cls.multiple_of)

355@resolves(pydantic.ConstrainedStr)
356def resolve_constr(cls): # type: ignore[no-untyped-def] # pragma: no cover
357 min_size = cls.min_length or 0
358 max_size = cls.max_length
359
360 if cls.regex is None and not cls.strip_whitespace:
361 return st.text(min_size=min_size, max_size=max_size)
362
363 if cls.regex is not None:
364 strategy = st.from_regex(cls.regex)
365 if cls.strip_whitespace:
366 strategy = strategy.filter(lambda s: s == s.strip())
367 elif cls.strip_whitespace:
368 repeats = '{{{},{}}}'.format(
369 min_size - 2 if min_size > 2 else 0,
370 max_size - 2 if (max_size or 0) > 2 else '',
371 )
372 if min_size >= 2:
373 strategy = st.from_regex(rf'\W.{repeats}\W')
374 elif min_size == 1:
375 strategy = st.from_regex(rf'\W(.{repeats}\W)?')
376 else:
377 assert min_size == 0
378 strategy = st.from_regex(rf'(\W(.{repeats}\W)?)?')
379
380 if min_size == 0 and max_size is None:
381 return strategy
382 elif max_size is None:
383 return strategy.filter(lambda s: min_size <= len(s))
384 return strategy.filter(lambda s: min_size <= len(s) <= max_size)
