159 async def process_request(self, path: str, request_headers: Headers) -> HTTPResponse | None:
160 """
161 This hook is called to determine if the websocket should return
162 an HTTP response and close.
163
164 Our behavior here is to start the ASGI application, and then wait
165 for either `accept` or `close` in order to determine if we should
166 close the connection.
167 """
168 path_portion, _, query_string = path.partition("?")
169
170 websockets.legacy.handshake.check_request(request_headers)
171
172 subprotocols: list[str] = []
173 for header in request_headers.get_all("Sec-WebSocket-Protocol"):
174 subprotocols.extend([token.strip() for token in header.split(",")])
175
176 asgi_headers = [
177 (name.encode("ascii"), value.encode("ascii", errors="surrogateescape"))
178 for name, value in request_headers.raw_items()
179 ]
180 path = unquote(path_portion)
181 full_path = self.root_path + path
182 full_raw_path = self.root_path.encode("ascii") + path_portion.encode("ascii")
183
184 self.scope = {
185 "type": "websocket",
186 "asgi": {"version": self.config.asgi_version, "spec_version": "2.4"},
187 "http_version": "1.1",
188 "scheme": self.scheme,
189 "server": self.server,
190 "client": self.client,
191 "root_path": self.root_path,
192 "path": full_path,
193 "raw_path": full_raw_path,
194 "query_string": query_string.encode("ascii"),
195 "headers": asgi_headers,
196 "subprotocols": subprotocols,
197 "state": self.app_state.copy(),
198 "extensions": {"websocket.http.response": {}},
199 }
200 task = self.loop.create_task(self.run_asgi())
201 task.add_done_callback(self.on_task_complete)
202 self.tasks.add(task)
203 await self.handshake_started_event.wait()
204 return self.initial_response

238 async def run_asgi(self) -> None:
239 """
240 Wrapper around the ASGI callable, handling exceptions and unexpected
241 termination states.
242 """
243 try:
244 result = await self.app(self.scope, self.asgi_receive, self.asgi_send) # type: ignore[func-returns-value]
245 except ClientDisconnected: # pragma: full coverage
246 self.closed_event.set()
247 self.transport.close()
248 except BaseException:
249 self.closed_event.set()
250 self.logger.exception("Exception in ASGI application\n")
251 if not self.handshake_started_event.is_set():
252 self.send_500_response()
253 else:
254 await self.handshake_completed_event.wait()
255 self.transport.close()
256 else:
257 self.closed_event.set()
258 if not self.handshake_started_event.is_set():
259 self.logger.error("ASGI callable returned without sending handshake.")
260 self.send_500_response()
261 self.transport.close()
262 elif result is not None:
263 self.logger.error("ASGI callable should return None, but returned '%s'.", result)

267 async def asgi_send(self, message: ASGISendEvent) -> None:
268 message_type = message["type"]
269
270 if not self.handshake_started_event.is_set():
271 if message_type == "websocket.accept":
272 message = cast("WebSocketAcceptEvent", message)
273 self.logger.info(
274 '%s - "WebSocket %s" [accepted]',
275 get_client_addr(self.scope),
276 get_path_with_query_string(self.scope),
277 )
278 self.initial_response = None
279 self.accepted_subprotocol = cast(Optional[Subprotocol], message.get("subprotocol"))
280 if "headers" in message:
281 self.extra_headers.extend(
282 # ASGI spec requires bytes
283 # But for compatibility we need to convert it to strings
284 (name.decode("latin-1"), value.decode("latin-1"))
285 for name, value in message["headers"]
286 )
287 self.handshake_started_event.set()
288
289 elif message_type == "websocket.close":
290 message = cast("WebSocketCloseEvent", message)
291 self.logger.info(
292 '%s - "WebSocket %s" 403',
293 get_client_addr(self.scope),
294 get_path_with_query_string(self.scope),
295 )
296 self.initial_response = (http.HTTPStatus.FORBIDDEN, [], b"")
297 self.handshake_started_event.set()
298 self.closed_event.set()
299
300 elif message_type == "websocket.http.response.start":
301 message = cast("WebSocketResponseStartEvent", message)
302 self.logger.info(
303 '%s - "WebSocket %s" %d',
304 get_client_addr(self.scope),
305 get_path_with_query_string(self.scope),
306 message["status"],
307 )
308 # websockets requires the status to be an enum. look it up.
309 status = http.HTTPStatus(message["status"])
310 headers = [
311 (name.decode("latin-1"), value.decode("latin-1")) for name, value in message.get("headers", [])
312 ]
313 self.initial_response = (status, headers, b"")
314 self.handshake_started_event.set()
315
316 else:
317 msg = (
318 "Expected ASGI message 'websocket.accept', 'websocket.close', "
319 "or 'websocket.http.response.start' but got '%s'."
320 )
321 raise RuntimeError(msg % message_type)
322
323 elif not self.closed_event.is_set() and self.initial_response is None:
324 await self.handshake_completed_event.wait()
325
326 try:
327 if message_type == "websocket.send":
328 message = cast("WebSocketSendEvent", message)
329 bytes_data = message.get("bytes")
330 text_data = message.get("text")
331 data = text_data if bytes_data is None else bytes_data
332 await self.send(data) # type: ignore[arg-type]
333
334 elif message_type == "websocket.close":
335 message = cast("WebSocketCloseEvent", message)
336 code = message.get("code", 1000)
337 reason = message.get("reason", "") or ""
338 await self.close(code, reason)
339 self.closed_event.set()
340
341 else:
342 msg = "Expected ASGI message 'websocket.send' or 'websocket.close', but got '%s'."
343 raise RuntimeError(msg % message_type)
344 except ConnectionClosed as exc:
345 raise ClientDisconnected from exc
346
347 elif self.initial_response is not None:
348 if message_type == "websocket.http.response.body":
349 message = cast("WebSocketResponseBodyEvent", message)
350 body = self.initial_response[2] + message["body"]
351 self.initial_response = self.initial_response[:2] + (body,)
352 if not message.get("more_body", False):
353 self.closed_event.set()
354 else:
355 msg = "Expected ASGI message 'websocket.http.response.body' but got '%s'."
356 raise RuntimeError(msg % message_type)
357
358 else:
359 msg = "Unexpected ASGI message '%s', after sending 'websocket.close' or response already completed."
360 raise RuntimeError(msg % message_type)

362 async def asgi_receive(self) -> WebSocketDisconnectEvent | WebSocketConnectEvent | WebSocketReceiveEvent:
363 if not self.connect_sent:
364 self.connect_sent = True
365 return {"type": "websocket.connect"}
366
367 await self.handshake_completed_event.wait()
368
369 if self.lost_connection_before_handshake:
370 # If the handshake failed or the app closed before handshake completion,
371 # use 1006 Abnormal Closure.
372 return {"type": "websocket.disconnect", "code": 1006}
373
374 if self.closed_event.is_set():
375 return {"type": "websocket.disconnect", "code": 1005}
376
377 try:
378 data = await self.recv()
379 except ConnectionClosed:
380 self.closed_event.set()
381 if self.ws_server.closing:
382 return {"type": "websocket.disconnect", "code": 1012}
383 return {"type": "websocket.disconnect", "code": self.close_code or 1005, "reason": self.close_reason}
384
385 if isinstance(data, str):
386 return {"type": "websocket.receive", "text": data}
387 return {"type": "websocket.receive", "bytes": data}
