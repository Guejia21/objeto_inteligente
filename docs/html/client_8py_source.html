407def _base62(
408 num: int,
409 base: str = string.digits + string.ascii_letters,
410 padding: int = 1,
411) -> str:
412 """Convert a number to base-62 representation."""
413 if num < 0:
414 raise ValueError("Number must be positive or zero")
415 digits = []
416 while num:
417 num, rest = divmod(num, 62)
418 digits.append(base[rest])
419 digits.extend(base[0] for _ in range(len(digits), padding))
420 return ''.join(reversed(digits))

486class MQTTMessageInfo:
487 """This is a class returned from `Client.publish()` and can be used to find
488 out the mid of the message that was published, and to determine whether the
489 message has been published, and/or wait until it is published.
490 """
491
492 __slots__ = 'mid', '_published', '_condition', 'rc', '_iterpos'
493
494 def __init__(self, mid: int):
495 self.mid = mid
496 """ The message Id (int)"""
497 self._published = False
498 self._condition = threading.Condition()
499 self.rc: MQTTErrorCode = MQTTErrorCode.MQTT_ERR_SUCCESS
500 """ The `MQTTErrorCode` that give status for this message.
501 This value could change until the message `is_published`"""
502 self._iterpos = 0
503
504 def __str__(self) -> str:
505 return str((self.rc, self.mid))
506
507 def __iter__(self) -> Iterator[MQTTErrorCode | int]:
508 self._iterpos = 0
509 return self
510
511 def __next__(self) -> MQTTErrorCode | int:
512 return self.next()
513
514 def next(self) -> MQTTErrorCode | int:
515 if self._iterpos == 0:
516 self._iterpos = 1
517 return self.rc
518 elif self._iterpos == 1:
519 self._iterpos = 2
520 return self.mid
521 else:
522 raise StopIteration
523
524 def __getitem__(self, index: int) -> MQTTErrorCode | int:
525 if index == 0:
526 return self.rc
527 elif index == 1:
528 return self.mid
529 else:
530 raise IndexError("index out of range")

532 def _set_as_published(self) -> None:
533 with self._condition:
534 self._published = True
535 self._condition.notify()

537 def wait_for_publish(self, timeout: float | None = None) -> None:
538 """Block until the message associated with this object is published, or
539 until the timeout occurs. If timeout is None, this will never time out.
540 Set timeout to a positive number of seconds, e.g. 1.2, to enable the
541 timeout.
542
543 :raises ValueError: if the message was not queued due to the outgoing
544 queue being full.
545
546 :raises RuntimeError: if the message was not published for another
547 reason.
548 """
549 if self.rc == MQTT_ERR_QUEUE_SIZE:
550 raise ValueError('Message is not queued due to ERR_QUEUE_SIZE')
551 elif self.rc == MQTT_ERR_AGAIN:
552 pass
553 elif self.rc > 0:
554 raise RuntimeError(f'Message publish failed: {error_string(self.rc)}')
555
556 timeout_time = None if timeout is None else time_func() + timeout
557 timeout_tenth = None if timeout is None else timeout / 10.
558 def timed_out() -> bool:
559 return False if timeout_time is None else time_func() > timeout_time
560
561 with self._condition:
562 while not self._published and not timed_out():
563 self._condition.wait(timeout_tenth)
564
565 if self.rc > 0:
566 raise RuntimeError(f'Message publish failed: {error_string(self.rc)}')
567
568 def is_published(self) -> bool:
569 """Returns True if the message associated with this object has been
570 published, else returns False.
571
572 To wait for this to become true, look at `wait_for_publish`.
573 """
574 if self.rc == MQTTErrorCode.MQTT_ERR_QUEUE_SIZE:
575 raise ValueError('Message is not queued due to ERR_QUEUE_SIZE')
576 elif self.rc == MQTTErrorCode.MQTT_ERR_AGAIN:
577 pass
578 elif self.rc > 0:
579 raise RuntimeError(f'Message publish failed: {error_string(self.rc)}')
580
581 with self._condition:
582 return self._published

633
634 This is the main class for use communicating with an MQTT broker.
635
636 General usage flow:
637
638 * Use `connect()`, `connect_async()` or `connect_srv()` to connect to a broker
639 * Use `loop_start()` to set a thread running to call `loop()` for you.
640 * Or use `loop_forever()` to handle calling `loop()` for you in a blocking function.
641 * Or call `loop()` frequently to maintain network traffic flow with the broker
642 * Use `subscribe()` to subscribe to a topic and receive messages
643 * Use `publish()` to send messages
644 * Use `disconnect()` to disconnect from the broker
645
646 Data returned from the broker is made available with the use of callback
647 functions as described below.
648
649 :param CallbackAPIVersion callback_api_version: define the API version for user-callback (on_connect, on_publish,...).
650 This field is required and it's recommended to use the latest version (CallbackAPIVersion.API_VERSION2).
651 See each callback for description of API for each version. The file docs/migrations.rst contains details on
652 how to migrate between version.
653

655 broker. If client_id is zero length or None, then the behaviour is
656 defined by which protocol version is in use. If using MQTT v3.1.1, then
657 a zero length client id will be sent to the broker and the broker will
658 generate a random for the client. If using MQTT v3.1 then an id will be
659 randomly generated. In both cases, clean_session must be True. If this
660 is not the case a ValueError will be raised.

665 subscription information and queued messages will be retained when the
666 client disconnects.

668 disconnect. Calling connect() or reconnect() will cause the messages to
669 be resent. Use reinitialise() to reset a client to its original state.

1606 def loop(self, timeout: float = 1.0) -> MQTTErrorCode:
1607 """Process network events.
1608
1609 It is strongly recommended that you use `loop_start()`, or
1610 `loop_forever()`, or if you are using an external event loop using
1611 `loop_read()`, `loop_write()`, and `loop_misc()`. Using loop() on it's own is
1612 no longer recommended.
1613
1614 This function must be called regularly to ensure communication with the
1615 broker is carried out. It calls select() on the network socket to wait
1616 for network events. If incoming data is present it will then be
1617 processed. Outgoing commands, from e.g. `publish()`, are normally sent
1618 immediately that their function is called, but this is not always
1619 possible. loop() will also attempt to send any remaining outgoing
1620 messages, which also includes commands that are part of the flow for
1621 messages with QoS>0.
1622
1623 :param int timeout: The time in seconds to wait for incoming/outgoing network
1624 traffic before timing out and returning.
1625
1626 Returns MQTT_ERR_SUCCESS on success.
1627 Returns >0 on error.
1628
1629 A ValueError will be raised if timeout < 0"""
1630
1631 if self._sockpairR is None or self._sockpairW is None:
1632 self._reset_sockets(sockpair_only=True)
1633 self._sockpairR, self._sockpairW = _socketpair_compat()
1634
1635 return self._loop(timeout)
1636
1637 def _loop(self, timeout: float = 1.0) -> MQTTErrorCode:
1638 if timeout < 0.0:
1639 raise ValueError('Invalid timeout.')
1640
1641 if self.want_write():
1642 wlist = [self._sock]
1643 else:
1644 wlist = []
1645
1646 # used to check if there are any bytes left in the (SSL) socket
1647 pending_bytes = 0
1648 if hasattr(self._sock, 'pending'):
1649 pending_bytes = self._sock.pending() # type: ignore[union-attr]
1650
1651 # if bytes are pending do not wait in select
1652 if pending_bytes > 0:
1653 timeout = 0.0
1654
1655 # sockpairR is used to break out of select() before the timeout, on a
1656 # call to publish() etc.
1657 if self._sockpairR is None:
1658 rlist = [self._sock]
1659 else:
1660 rlist = [self._sock, self._sockpairR]
1661
1662 try:
1663 socklist = select.select(rlist, wlist, [], timeout)
1664 except TypeError:
1665 # Socket isn't correct type, in likelihood connection is lost
1666 # ... or we called disconnect(). In that case the socket will
1667 # be closed but some loop (like loop_forever) will continue to
1668 # call _loop(). We still want to break that loop by returning an
1669 # rc != MQTT_ERR_SUCCESS and we don't want state to change from
1670 # mqtt_cs_disconnecting.
1671 if self._state not in (_ConnectionState.MQTT_CS_DISCONNECTING, _ConnectionState.MQTT_CS_DISCONNECTED):
1672 self._state = _ConnectionState.MQTT_CS_CONNECTION_LOST
1673 return MQTTErrorCode.MQTT_ERR_CONN_LOST
1674 except ValueError:
1675 # Can occur if we just reconnected but rlist/wlist contain a -1 for
1676 # some reason.
1677 if self._state not in (_ConnectionState.MQTT_CS_DISCONNECTING, _ConnectionState.MQTT_CS_DISCONNECTED):
1678 self._state = _ConnectionState.MQTT_CS_CONNECTION_LOST
1679 return MQTTErrorCode.MQTT_ERR_CONN_LOST
1680 except Exception:
1681 # Note that KeyboardInterrupt, etc. can still terminate since they
1682 # are not derived from Exception
1683 return MQTTErrorCode.MQTT_ERR_UNKNOWN
1684
1685 if self._sock in socklist[0] or pending_bytes > 0:
1686 rc = self.loop_read()
1687 if rc or self._sock is None:
1688 return rc
1689
1690 if self._sockpairR and self._sockpairR in socklist[0]:
1691 # Stimulate output write even though we didn't ask for it, because
1692 # at that point the publish or other command wasn't present.
1693 socklist[1].insert(0, self._sock)
1694 # Clear sockpairR - only ever a single byte written.
1695 try:
1696 # Read many bytes at once - this allows up to 10000 calls to
1697 # publish() inbetween calls to loop().
1698 self._sockpairR.recv(10000)
1699 except BlockingIOError:
1700 pass
1701
1702 if self._sock in socklist[1]:
1703 rc = self.loop_write()
1704 if rc or self._sock is None:
1705 return rc
1706
1707 return self.loop_misc()

1709 def publish(
1710 self,
1711 topic: str,
1712 payload: PayloadType = None,
1713 qos: int = 0,
1714 retain: bool = False,
1715 properties: Properties | None = None,
1716 ) -> MQTTMessageInfo:
1717 """Publish a message on a topic.
1718
1719 This causes a message to be sent to the broker and subsequently from
1720 the broker to any clients subscribing to matching topics.
1721
1722 :param str topic: The topic that the message should be published on.
1723 :param payload: The actual message to send. If not given, or set to None a
1724 zero length message will be used. Passing an int or float will result
1725 in the payload being converted to a string representing that number. If
1726 you wish to send a true int/float, use struct.pack() to create the
1727 payload you require.
1728 :param int qos: The quality of service level to use.
1729 :param bool retain: If set to true, the message will be set as the "last known
1730 good"/retained message for the topic.
1731 :param Properties properties: (MQTT v5.0 only) the MQTT v5.0 properties to be included.
1732
1733 Returns a `MQTTMessageInfo` class, which can be used to determine whether
1734 the message has been delivered (using `is_published()`) or to block
1735 waiting for the message to be delivered (`wait_for_publish()`). The
1736 message ID and return code of the publish() call can be found at
1737 :py:attr:`info.mid <MQTTMessage.mid>` and :py:attr:`info.rc <MQTTMessage.rc>`.
1738
1739 For backwards compatibility, the `MQTTMessageInfo` class is iterable so
1740 the old construct of ``(rc, mid) = client.publish(...)`` is still valid.
1741
1742 rc is MQTT_ERR_SUCCESS to indicate success or MQTT_ERR_NO_CONN if the
1743 client is not currently connected. mid is the message ID for the
1744 publish request. The mid value can be used to track the publish request
1745 by checking against the mid argument in the on_publish() callback if it
1746 is defined.
1747
1748 :raises ValueError: if topic is None, has zero length or is
1749 invalid (contains a wildcard), except if the MQTT version used is v5.0.
1750 For v5.0, a zero length topic can be used when a Topic Alias has been set.
1751 :raises ValueError: if qos is not one of 0, 1 or 2
1752 :raises ValueError: if the length of the payload is greater than 268435455 bytes.
1753 """
1754 if self._protocol != MQTTv5:
1755 if topic is None or len(topic) == 0:
1756 raise ValueError('Invalid topic.')
1757
1758 topic_bytes = topic.encode('utf-8')
1759
1760 self._raise_for_invalid_topic(topic_bytes)
1761
1762 if qos < 0 or qos > 2:
1763 raise ValueError('Invalid QoS level.')
1764
1765 local_payload = _encode_payload(payload)
1766
1767 if len(local_payload) > 268435455:
1768 raise ValueError('Payload too large.')
1769
1770 local_mid = self._mid_generate()
1771
1772 if qos == 0:
1773 info = MQTTMessageInfo(local_mid)
1774 rc = self._send_publish(
1775 local_mid, topic_bytes, local_payload, qos, retain, False, info, properties)
1776 info.rc = rc
1777 return info
1778 else:
1779 message = MQTTMessage(local_mid, topic_bytes)
1780 message.timestamp = time_func()
1781 message.payload = local_payload
1782 message.qos = qos
1783 message.retain = retain
1784 message.dup = False
1785 message.properties = properties
1786
1787 with self._out_message_mutex:
1788 if self._max_queued_messages > 0 and len(self._out_messages) >= self._max_queued_messages:
1789 message.info.rc = MQTTErrorCode.MQTT_ERR_QUEUE_SIZE
1790 return message.info
1791
1792 if local_mid in self._out_messages:
1793 message.info.rc = MQTTErrorCode.MQTT_ERR_QUEUE_SIZE
1794 return message.info
1795
1796 self._out_messages[message.mid] = message
1797 if self._max_inflight_messages == 0 or self._inflight_messages < self._max_inflight_messages:
1798 self._inflight_messages += 1
1799 if qos == 1:
1800 message.state = mqtt_ms_wait_for_puback
1801 elif qos == 2:
1802 message.state = mqtt_ms_wait_for_pubrec
1803
1804 rc = self._send_publish(message.mid, topic_bytes, message.payload, message.qos, message.retain,
1805 message.dup, message.info, message.properties)
1806
1807 # remove from inflight messages so it will be send after a connection is made
1808 if rc == MQTTErrorCode.MQTT_ERR_NO_CONN:
1809 self._inflight_messages -= 1
1810 message.state = mqtt_ms_publish
1811
1812 message.info.rc = rc
1813 return message.info
1814 else:
1815 message.state = mqtt_ms_queued
1816 message.info.rc = MQTTErrorCode.MQTT_ERR_SUCCESS
1817 return message.info

3054 def _packet_read(self) -> MQTTErrorCode:

3056 # available - ie. at least one byte. What we do depends on what data we
3057 # already have.
3058 # If we've not got a command, attempt to read one and save it. This should
3059 # always work because it's only a single byte.
3060 # Then try to read the remaining length. This may fail because it is may
3061 # be more than one byte - will need to save data pending next read if it
3062 # does fail.
3063 # Then try to read the remaining payload, where 'payload' here means the
3064 # combined variable header and actual payload. This is the most likely to
3065 # fail due to longer length, so save current data and current position.
3066 # After all data is read, send to _mqtt_handle_packet() to deal with.
3067 # Finally, free the memory and reset everything to starting conditions.

3070 command = self._sock_recv(1)
3071 except BlockingIOError:

3073 except TimeoutError as err:

3105 # Anything more likely means a broken/malicious client.
3106 if len(self._in_packet['remaining_count']) > 4:
