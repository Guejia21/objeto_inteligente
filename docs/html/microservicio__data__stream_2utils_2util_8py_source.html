8 Args:

11 Returns:

17
18 # Construir objeto simplificado

20 "object": {

23 "description": dic_obj.get("description", ""),

27 "datastreams": []
28 }
29
30 # Convertir datastreams
31 for rec in dic_rec:
32 datastream = {
33 "datastream_id": rec.get("datastream_id", ""),
34 "datastream_format": rec.get("datastream_format", ""),
35 "datastream_type": rec.get("datastream_type", ""),
36 "unit_symbol": rec.get("symbol", "")
37 }

39 # Agregar campos opcionales solo si existen
