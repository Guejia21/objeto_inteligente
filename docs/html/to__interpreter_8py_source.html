172 """
173 Call the given function with the given arguments in a subinterpreter.
174
175 .. warning:: On Python 3.13, the :mod:`concurrent.interpreters` module was not yet
176 available, so the code path for that Python version relies on an undocumented,
177 private API. As such, it is recommended to not rely on this function for anything
178 mission-critical on Python 3.13.
179
180 :param func: a callable
181 :param args: the positional arguments for the callable
182 :param limiter: capacity limiter to use to limit the total number of subinterpreters
183 running (if omitted, the default limiter is used)
184 :return: the result of the call
185 :raises BrokenWorkerInterpreter: if there's an internal error in a subinterpreter
186
187 """
188 if limiter is None:
189 limiter = current_default_interpreter_limiter()
190
191 try:
192 idle_workers = _idle_workers.get()
193 except LookupError:
194 idle_workers = deque()
195 _idle_workers.set(idle_workers)
196 atexit.register(_stop_workers, idle_workers)
197
198 async with limiter:
199 try:
200 worker = idle_workers.pop()
201 except IndexError:
202 worker = Worker()
203
204 try:
205 return await to_thread.run_sync(
206 worker.call,
207 func,
208 args,
209 limiter=limiter,
210 )
211 finally:
212 # Prune workers that have been idle for too long
213 now = current_time()
214 while idle_workers:
215 if now - idle_workers[0].last_used <= MAX_WORKER_IDLE_TIME:
216 break
217
218 await to_thread.run_sync(idle_workers.popleft().destroy, limiter=limiter)
219
220 worker.last_used = current_time()
221 idle_workers.append(worker)
