306 def expand_type(
307 self,
308 current_info: TypeInfo,
309 api: SemanticAnalyzerPluginInterface,
310 force_typevars_invariant: bool = False,
311 include_root_type: bool = False,
312 ) -> Type | None:
313 """Based on mypy.plugins.dataclasses.DataclassAttribute.expand_type."""
314 if force_typevars_invariant:
315 # In some cases, mypy will emit an error "Cannot use a covariant type variable as a parameter"
316 # To prevent that, we add an option to replace typevars with invariant ones while building certain
317 # method signatures (in particular, `__init__`). There may be a better way to do this, if this causes
318 # us problems in the future, we should look into why the dataclasses plugin doesn't have this issue.
319 if isinstance(self.type, TypeVarType):
320 modified_type = self.type.copy_modified()
321 modified_type.variance = INVARIANT
322 self.type = modified_type
323
324 if self.type is not None and self.info.self_type is not None:
325 # In general, it is not safe to call `expand_type()` during semantic analysis,
326 # however this plugin is called very late, so all types should be fully ready.
327 # Also, it is tricky to avoid eager expansion of Self types here (e.g. because
328 # we serialize attributes).
329 with state.strict_optional_set(api.options.strict_optional):
330 filled_with_typevars = fill_typevars(current_info)
331 # Cannot be TupleType as current_info represents a Pydantic model:
332 assert isinstance(filled_with_typevars, Instance)
333 if force_typevars_invariant:
334 for arg in filled_with_typevars.args:
335 if isinstance(arg, TypeVarType):
336 arg.variance = INVARIANT
337
338 expanded_type = expand_type(self.type, {self.info.self_type.id: filled_with_typevars})
339 if include_root_type and isinstance(expanded_type, Instance) and is_root_model(expanded_type.type):
340 # When a root model is used as a field, Pydantic allows both an instance of the root model
341 # as well as instances of the `root` field type:
342 root_type = expanded_type.type['root'].type
343 if root_type is None:
344 # Happens if the hint for 'root' has unsolved forward references
345 return expanded_type
346 expanded_root_type = expand_type_by_instance(root_type, expanded_type)
347 expanded_type = UnionType([expanded_type, expanded_root_type])
348 return expanded_type
349 return self.type

453 def transform(self) -> bool:
454 """Configures the BaseModel subclass according to the plugin settings.
455
456 In particular:
457
458 * determines the model config and fields,
459 * adds a fields-aware signature for the initializer and construct methods
460 * freezes the class if frozen = True
461 * stores the fields, config, and if the class is settings in the mypy metadata for access by subclasses
462 """
463 info = self._cls.info
464 is_a_root_model = is_root_model(info)
465 config = self.collect_config()
466 fields, class_vars = self.collect_fields_and_class_vars(config, is_a_root_model)
467 if fields is None or class_vars is None:
468 # Some definitions are not ready. We need another pass.
469 return False
470 for field in fields:
471 if field.type is None:
472 return False
473
474 is_settings = info.has_base(BASESETTINGS_FULLNAME)
475 self.add_initializer(fields, config, is_settings, is_a_root_model)
476 self.add_model_construct_method(fields, config, is_settings, is_a_root_model)
477 self.set_frozen(fields, self._api, frozen=config.frozen is True)
478
479 self.adjust_decorator_signatures()
480
481 info.metadata[METADATA_KEY] = {
482 'fields': {field.name: field.serialize() for field in fields},
483 'class_vars': {class_var.name: class_var.serialize() for class_var in class_vars},
484 'config': config.get_values_dict(),
485 }
486
487 return True

489 def adjust_decorator_signatures(self) -> None:
490 """When we decorate a function `f` with `pydantic.validator(...)`, `pydantic.field_validator`
491 or `pydantic.serializer(...)`, mypy sees `f` as a regular method taking a `self` instance,
492 even though pydantic internally wraps `f` with `classmethod` if necessary.
493
494 Teach mypy this by marking any function whose outermost decorator is a `validator()`,
495 `field_validator()` or `serializer()` call as a `classmethod`.
496 """
497 for sym in self._cls.info.names.values():
498 if isinstance(sym.node, Decorator):
499 first_dec = sym.node.original_decorators[0]
500 if (
501 isinstance(first_dec, CallExpr)
502 and isinstance(first_dec.callee, NameExpr)
503 and first_dec.callee.fullname in IMPLICIT_CLASSMETHOD_DECORATOR_FULLNAMES
504 # @model_validator(mode="after") is an exception, it expects a regular method
505 and not (
506 first_dec.callee.fullname == MODEL_VALIDATOR_FULLNAME
507 and any(
508 first_dec.arg_names[i] == 'mode' and isinstance(arg, StrExpr) and arg.value == 'after'
509 for i, arg in enumerate(first_dec.args)
510 )
511 )
512 ):
513 # TODO: Only do this if the first argument of the decorated function is `cls`
514 sym.node.func.is_class = True

516 def collect_config(self) -> ModelConfigData: # noqa: C901 (ignore complexity)
517 """Collects the values of the config attributes that are used by the plugin, accounting for parent classes."""
518 cls = self._cls
519 config = ModelConfigData()
520
521 has_config_kwargs = False
522 has_config_from_namespace = False
523
524 # Handle `class MyModel(BaseModel, <name>=<expr>, ...):`
525 for name, expr in cls.keywords.items():
526 config_data = self.get_config_update(name, expr)
527 if config_data:
528 has_config_kwargs = True
529 config.update(config_data)
530
531 # Handle `model_config`
532 stmt: Statement | None = None
533 for stmt in cls.defs.body:
534 if not isinstance(stmt, (AssignmentStmt, ClassDef)):
535 continue
536
537 if isinstance(stmt, AssignmentStmt):
538 lhs = stmt.lvalues[0]
539 if not isinstance(lhs, NameExpr) or lhs.name != 'model_config':
540 continue
541
542 if isinstance(stmt.rvalue, CallExpr): # calls to `dict` or `ConfigDict`
543 for arg_name, arg in zip(stmt.rvalue.arg_names, stmt.rvalue.args):
544 if arg_name is None:
545 continue
546 config.update(self.get_config_update(arg_name, arg, lax_extra=True))
547 elif isinstance(stmt.rvalue, DictExpr): # dict literals
548 for key_expr, value_expr in stmt.rvalue.items:
549 if not isinstance(key_expr, StrExpr):
550 continue
551 config.update(self.get_config_update(key_expr.value, value_expr))
552
553 elif isinstance(stmt, ClassDef):
554 if stmt.name != 'Config': # 'deprecated' Config-class
555 continue
556 for substmt in stmt.defs.body:
557 if not isinstance(substmt, AssignmentStmt):
558 continue
559 lhs = substmt.lvalues[0]
560 if not isinstance(lhs, NameExpr):
561 continue
562 config.update(self.get_config_update(lhs.name, substmt.rvalue))
563
564 if has_config_kwargs:
565 self._api.fail(
566 'Specifying config in two places is ambiguous, use either Config attribute or class kwargs',
567 cls,
568 )
569 break
570
571 has_config_from_namespace = True
572
573 if has_config_kwargs or has_config_from_namespace:
574 if (
575 stmt
576 and config.has_alias_generator
577 and not (config.validate_by_name or config.populate_by_name)
578 and self.plugin_config.warn_required_dynamic_aliases
579 ):
580 error_required_dynamic_aliases(self._api, stmt)
581
582 for info in cls.info.mro[1:]: # 0 is the current class
583 if METADATA_KEY not in info.metadata:
584 continue
585
586 # Each class depends on the set of fields in its ancestors
587 self._api.add_plugin_dependency(make_wildcard_trigger(info.fullname))
588 for name, value in info.metadata[METADATA_KEY]['config'].items():
589 config.setdefault(name, value)
590 return config

592 def collect_fields_and_class_vars(
593 self, model_config: ModelConfigData, is_root_model: bool
594 ) -> tuple[list[PydanticModelField] | None, list[PydanticModelClassVar] | None]:
595 """Collects the fields for the model, accounting for parent classes."""
596 cls = self._cls
597
598 # First, collect fields and ClassVars belonging to any class in the MRO, ignoring duplicates.
599 #
600 # We iterate through the MRO in reverse because attrs defined in the parent must appear
601 # earlier in the attributes list than attrs defined in the child. See:
602 # https://docs.python.org/3/library/dataclasses.html#inheritance
603 #
604 # However, we also want fields defined in the subtype to override ones defined
605 # in the parent. We can implement this via a dict without disrupting the attr order
606 # because dicts preserve insertion order in Python 3.7+.
607 found_fields: dict[str, PydanticModelField] = {}
608 found_class_vars: dict[str, PydanticModelClassVar] = {}
609 for info in reversed(cls.info.mro[1:-1]): # 0 is the current class, -2 is BaseModel, -1 is object
610 # if BASEMODEL_METADATA_TAG_KEY in info.metadata and BASEMODEL_METADATA_KEY not in info.metadata:
611 # # We haven't processed the base class yet. Need another pass.
612 # return None, None
613 if METADATA_KEY not in info.metadata:
614 continue
615
616 # Each class depends on the set of attributes in its dataclass ancestors.
617 self._api.add_plugin_dependency(make_wildcard_trigger(info.fullname))
618
619 for name, data in info.metadata[METADATA_KEY]['fields'].items():
620 field = PydanticModelField.deserialize(info, data, self._api)
621 # (The following comment comes directly from the dataclasses plugin)
622 # TODO: We shouldn't be performing type operations during the main
623 # semantic analysis pass, since some TypeInfo attributes might
624 # still be in flux. This should be performed in a later phase.
625 field.expand_typevar_from_subtype(cls.info, self._api)
626 found_fields[name] = field
627
628 sym_node = cls.info.names.get(name)
629 if sym_node and sym_node.node and not isinstance(sym_node.node, Var):
630 self._api.fail(
631 'BaseModel field may only be overridden by another field',
632 sym_node.node,
633 )
634 # Collect ClassVars
635 for name, data in info.metadata[METADATA_KEY]['class_vars'].items():
636 found_class_vars[name] = PydanticModelClassVar.deserialize(data)
637
638 # Second, collect fields and ClassVars belonging to the current class.
639 current_field_names: set[str] = set()
640 current_class_vars_names: set[str] = set()
641 for stmt in self._get_assignment_statements_from_block(cls.defs):
642 maybe_field = self.collect_field_or_class_var_from_stmt(stmt, model_config, found_class_vars)
643 if maybe_field is None:
644 continue
645
646 lhs = stmt.lvalues[0]
647 assert isinstance(lhs, NameExpr) # collect_field_or_class_var_from_stmt guarantees this
648 if isinstance(maybe_field, PydanticModelField):
649 if is_root_model and lhs.name != 'root':
650 error_extra_fields_on_root_model(self._api, stmt)
651 else:
652 current_field_names.add(lhs.name)

674 def collect_field_or_class_var_from_stmt( # noqa C901
675 self, stmt: AssignmentStmt, model_config: ModelConfigData, class_vars: dict[str, PydanticModelClassVar]
676 ) -> PydanticModelField | PydanticModelClassVar | None:
677 """Get pydantic model field from statement.
678
679 Args:
680 stmt: The statement.
681 model_config: Configuration settings for the model.
682 class_vars: ClassVars already known to be defined on the model.
683
684 Returns:
685 A pydantic model field if it could find the field in statement. Otherwise, `None`.
686 """
687 cls = self._cls
688
689 lhs = stmt.lvalues[0]
690 if not isinstance(lhs, NameExpr) or not _fields.is_valid_field_name(lhs.name) or lhs.name == 'model_config':
691 return None
692
693 if not stmt.new_syntax:
694 if (
695 isinstance(stmt.rvalue, CallExpr)
696 and isinstance(stmt.rvalue.callee, CallExpr)
697 and isinstance(stmt.rvalue.callee.callee, NameExpr)
698 and stmt.rvalue.callee.callee.fullname in DECORATOR_FULLNAMES
699 ):
700 # This is a (possibly-reused) validator or serializer, not a field
701 # In particular, it looks something like: my_validator = validator('my_field')(f)
702 # Eventually, we may want to attempt to respect model_config['ignored_types']
703 return None
704
705 if lhs.name in class_vars:
706 # Class vars are not fields and are not required to be annotated
707 return None
708
709 # The assignment does not have an annotation, and it's not anything else we recognize
710 error_untyped_fields(self._api, stmt)
711 return None
712
713 lhs = stmt.lvalues[0]
714 if not isinstance(lhs, NameExpr):
715 return None
716
717 if not _fields.is_valid_field_name(lhs.name) or lhs.name == 'model_config':
718 return None
719
720 sym = cls.info.names.get(lhs.name)
721 if sym is None: # pragma: no cover
722 # This is likely due to a star import (see the dataclasses plugin for a more detailed explanation)
723 # This is the same logic used in the dataclasses plugin
724 return None
725
726 node = sym.node
727 if isinstance(node, PlaceholderNode): # pragma: no cover
728 # See the PlaceholderNode docstring for more detail about how this can occur
729 # Basically, it is an edge case when dealing with complex import logic
730
731 # The dataclasses plugin now asserts this cannot happen, but I'd rather not error if it does..
732 return None
733
734 if isinstance(node, TypeAlias):
735 self._api.fail(
736 'Type aliases inside BaseModel definitions are not supported at runtime',
737 node,
738 )
739 # Skip processing this node. This doesn't match the runtime behaviour,
740 # but the only alternative would be to modify the SymbolTable,
741 # and it's a little hairy to do that in a plugin.
742 return None
743
744 if not isinstance(node, Var): # pragma: no cover
745 # Don't know if this edge case still happens with the `is_valid_field` check above
746 # but better safe than sorry
747
748 # The dataclasses plugin now asserts this cannot happen, but I'd rather not error if it does..
749 return None
750
751 # x: ClassVar[int] is not a field
752 if node.is_classvar:
753 return PydanticModelClassVar(lhs.name)
754
755 # x: InitVar[int] is not supported in BaseModel
756 node_type = get_proper_type(node.type)
757 if isinstance(node_type, Instance) and node_type.type.fullname == 'dataclasses.InitVar':
758 self._api.fail(
759 'InitVar is not supported in BaseModel',
760 node,
761 )
762
763 has_default = self.get_has_default(stmt)
764 strict = self.get_strict(stmt)
765
766 if sym.type is None and node.is_final and node.is_inferred:
767 # This follows the logic from the dataclasses plugin. The following comment is taken verbatim:
768 #
769 # This is a special case, assignment like x: Final = 42 is classified
770 # annotated above, but mypy strips the `Final` turning it into x = 42.
771 # We do not support inferred types in dataclasses, so we can try inferring
772 # type for simple literals, and otherwise require an explicit type
773 # argument for Final[...].
774 typ = self._api.analyze_simple_literal_type(stmt.rvalue, is_final=True)
775 if typ:
776 node.type = typ
777 else:
778 self._api.fail(
779 'Need type argument for Final[...] with non-literal default in BaseModel',
780 stmt,
781 )
782 node.type = AnyType(TypeOfAny.from_error)
783
784 if node.is_final and has_default:
785 # TODO this path should be removed (see https://github.com/pydantic/pydantic/issues/11119)
786 return PydanticModelClassVar(lhs.name)
787
788 alias, has_dynamic_alias = self.get_alias_info(stmt)
789 if (
790 has_dynamic_alias
791 and not (model_config.validate_by_name or model_config.populate_by_name)
792 and self.plugin_config.warn_required_dynamic_aliases
793 ):
794 error_required_dynamic_aliases(self._api, stmt)
795 is_frozen = self.is_field_frozen(stmt)
796
797 init_type = self._infer_dataclass_attr_init_type(sym, lhs.name, stmt)
798 return PydanticModelField(
799 name=lhs.name,
800 has_dynamic_alias=has_dynamic_alias,
801 has_default=has_default,
802 strict=strict,
803 alias=alias,
804 is_frozen=is_frozen,
805 line=stmt.line,
806 column=stmt.column,
807 type=init_type,
808 info=cls.info,
809 )

811 def _infer_dataclass_attr_init_type(self, sym: SymbolTableNode, name: str, context: Context) -> Type | None:
812 """Infer __init__ argument type for an attribute.
813
814 In particular, possibly use the signature of __set__.
815 """
816 default = sym.type
817 if sym.implicit:
818 return default
819 t = get_proper_type(sym.type)
820
821 # Perform a simple-minded inference from the signature of __set__, if present.
822 # We can't use mypy.checkmember here, since this plugin runs before type checking.
823 # We only support some basic scanerios here, which is hopefully sufficient for
824 # the vast majority of use cases.
825 if not isinstance(t, Instance):
826 return default
827 setter = t.type.get('__set__')
828 if setter:
829 if isinstance(setter.node, FuncDef):
830 super_info = t.type.get_containing_type_info('__set__')
831 assert super_info
832 if setter.type:
833 setter_type = get_proper_type(map_type_from_supertype(setter.type, t.type, super_info))
834 else:
835 return AnyType(TypeOfAny.unannotated)
836 if isinstance(setter_type, CallableType) and setter_type.arg_kinds == [
837 ARG_POS,
838 ARG_POS,
839 ARG_POS,
840 ]:
841 return expand_type_by_instance(setter_type.arg_types[2], t)
842 else:
843 self._api.fail(f'Unsupported signature for "__set__" in "{t.type.name}"', context)
844 else:
845 self._api.fail(f'Unsupported "__set__" in "{t.type.name}"', context)
846
847 return default

853

855 """
856 if '__init__' in self._cls.info.names and not self._cls.info.names['__init__'].plugin_generated:
857 return # Don't generate an __init__ if one already exists
858
859 typed = self.plugin_config.init_typed

864 fields,
865 typed=typed,
866 model_strict=model_strict,
867 requires_dynamic_aliases=requires_dynamic_aliases,

869 is_settings=is_settings,
870 is_root_model=is_root_model,

874 if is_settings:
875 base_settings_node = self._api.lookup_fully_qualified(BASESETTINGS_FULLNAME).node
876 assert isinstance(base_settings_node, TypeInfo)
877 if '__init__' in base_settings_node.names:

879 assert isinstance(base_settings_init_node, FuncDef)

1003 @staticmethod
1004 def get_has_default(stmt: AssignmentStmt) -> bool:
1005 """Returns a boolean indicating whether the field defined in `stmt` is a required field."""
1006 expr = stmt.rvalue
1007 if isinstance(expr, TempNode):
1008 # TempNode means annotation-only, so has no default
1009 return False
1010 if isinstance(expr, CallExpr) and isinstance(expr.callee, RefExpr) and expr.callee.fullname == FIELD_FULLNAME:
1011 # The "default value" is a call to `Field`; at this point, the field has a default if and only if:
1012 # * there is a positional argument that is not `...`
1013 # * there is a keyword argument named "default" that is not `...`
1014 # * there is a "default_factory" that is not `None`
1015 for arg, name in zip(expr.args, expr.arg_names):
1016 # If name is None, then this arg is the default because it is the only positional argument.
1017 if name is None or name == 'default':
1018 return arg.__class__ is not EllipsisExpr
1019 if name == 'default_factory':
1020 return not (isinstance(arg, NameExpr) and arg.fullname == 'builtins.None')
1021 return False
1022 # Has no default if the "default value" is Ellipsis (i.e., `field_name: Annotation = ...`)
1023 return not isinstance(expr, EllipsisExpr)

1266def add_method(
1267 api: SemanticAnalyzerPluginInterface | CheckerPluginInterface,
1268 cls: ClassDef,
1269 name: str,
1270 args: list[Argument],
1271 return_type: Type,
1272 self_type: Type | None = None,
1273 tvar_def: TypeVarType | None = None,
1274 is_classmethod: bool = False,
1275) -> None:
1276 """Very closely related to `mypy.plugins.common.add_method_to_class`, with a few pydantic-specific changes."""
1277 info = cls.info
1278
1279 # First remove any previously generated methods with the same name
1280 # to avoid clashes and problems in the semantic analyzer.
1281 if name in info.names:
1282 sym = info.names[name]
1283 if sym.plugin_generated and isinstance(sym.node, FuncDef):
1284 cls.defs.body.remove(sym.node) # pragma: no cover
1285
1286 if isinstance(api, SemanticAnalyzerPluginInterface):
1287 function_type = api.named_type('builtins.function')
1288 else:
1289 function_type = api.named_generic_type('builtins.function', [])
1290
1291 if is_classmethod:
1292 self_type = self_type or TypeType(fill_typevars(info))
1293 first = [Argument(Var('_cls'), self_type, None, ARG_POS, True)]
1294 else:
1295 self_type = self_type or fill_typevars(info)
1296 # `self` is positional *ONLY* here, but this can't be expressed
1297 # fully in the mypy internal API. ARG_POS is the closest we can get.
1298 # Using ARG_POS will, however, give mypy errors if a `self` field
1299 # is present on a model:
1300 #
1301 # Name "self" already defined (possibly by an import) [no-redef]
1302 #
1303 # As a workaround, we give this argument a name that will
1304 # never conflict. By its positional nature, this name will not
1305 # be used or exposed to users.
1306 first = [Argument(Var('__pydantic_self__'), self_type, None, ARG_POS)]
1307 args = first + args
1308
1309 arg_types, arg_names, arg_kinds = [], [], []
1310 for arg in args:
1311 assert arg.type_annotation, 'All arguments must be fully typed.'
1312 arg_types.append(arg.type_annotation)
1313 arg_names.append(arg.variable.name)
1314 arg_kinds.append(arg.kind)
1315
1316 signature = CallableType(
1317 arg_types, arg_kinds, arg_names, return_type, function_type, variables=[tvar_def] if tvar_def else None
1318 )
1319
1320 func = FuncDef(name, args, Block([PassStmt()]))
1321 func.info = info
1322 func.type = set_callable_name(signature, func)
1323 func.is_class = is_classmethod
1324 func._fullname = info.fullname + '.' + name
1325 func.line = info.line
1326
1327 # NOTE: we would like the plugin generated node to dominate, but we still
1328 # need to keep any existing definitions so they get semantically analyzed.
1329 if name in info.names:
1330 # Get a nice unique name instead.
1331 r_name = get_unique_redefinition_name(name, info.names)
1332 info.names[r_name] = info.names[name]
1333
1334 # Add decorator for is_classmethod
1335 # The dataclasses plugin claims this is unnecessary for classmethods, but not including it results in a
1336 # signature incompatible with the superclass, which causes mypy errors to occur for every subclass of BaseModel.
1337 if is_classmethod:
1338 func.is_decorated = True
1339 v = Var(name, func.type)
1340 v.info = info
1341 v._fullname = func._fullname
1342 v.is_classmethod = True
1343 dec = Decorator(func, [NameExpr('classmethod')], v)
1344 dec.line = info.line
1345 sym = SymbolTableNode(MDEF, dec)
1346 else:
1347 sym = SymbolTableNode(MDEF, func)
1348 sym.plugin_generated = True
1349 info.names[name] = sym
1350
1351 info.defn.defs.body.append(func)
