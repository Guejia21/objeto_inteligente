44def _getattr_no_parents(obj: Any, attribute: str) -> Any:
45 """Returns the attribute value without attempting to look up attributes from parent types."""
46 if hasattr(obj, '__dict__'):
47 try:
48 return obj.__dict__[attribute]
49 except KeyError:
50 pass
51
52 slots = getattr(obj, '__slots__', None)
53 if slots is not None and attribute in slots:
54 return getattr(obj, attribute)
55 else:
56 raise AttributeError(attribute)

69@final
70class TypeAdapter(Generic[T]):
71 """!!! abstract "Usage Documentation"
72 [`TypeAdapter`](../concepts/type_adapter.md)
73
74 Type adapters provide a flexible way to perform validation and serialization based on a Python type.
75
76 A `TypeAdapter` instance exposes some of the functionality from `BaseModel` instance methods
77 for types that do not have such methods (such as dataclasses, primitive types, and more).
78
79 **Note:** `TypeAdapter` instances are not types, and cannot be used as type annotations for fields.
80
81 Args:
82 type: The type associated with the `TypeAdapter`.
83 config: Configuration for the `TypeAdapter`, should be a dictionary conforming to
84 [`ConfigDict`][pydantic.config.ConfigDict].
85
86 !!! note
87 You cannot provide a configuration when instantiating a `TypeAdapter` if the type you're using
88 has its own config that cannot be overridden (ex: `BaseModel`, `TypedDict`, and `dataclass`). A
89 [`type-adapter-config-unused`](../errors/usage_errors.md#type-adapter-config-unused) error will
90 be raised in this case.
91 _parent_depth: Depth at which to search for the [parent frame][frame-objects]. This frame is used when
92 resolving forward annotations during schema building, by looking for the globals and locals of this
93 frame. Defaults to 2, which will result in the frame where the `TypeAdapter` was instantiated.
94
95 !!! note
96 This parameter is named with an underscore to suggest its private nature and discourage use.
97 It may be deprecated in a minor version, so we only recommend using it if you're comfortable
98 with potential change in behavior/support. It's default value is 2 because internally,
99 the `TypeAdapter` class makes another call to fetch the frame.
100 module: The module that passes to plugin if provided.
101
102 Attributes:
103 core_schema: The core schema for the type.
104 validator: The schema validator for the type.
105 serializer: The schema serializer for the type.
106 pydantic_complete: Whether the core schema for the type is successfully built.
107
108 ??? tip "Compatibility with `mypy`"
109 Depending on the type used, `mypy` might raise an error when instantiating a `TypeAdapter`. As a workaround, you can explicitly
110 annotate your variable:
111
112 ```py
113 from typing import Union
114
115 from pydantic import TypeAdapter
116
117 ta: TypeAdapter[Union[str, int]] = TypeAdapter(Union[str, int]) # type: ignore[arg-type]
118 ```
119
120 ??? info "Namespace management nuances and implementation details"
121
122 Here, we collect some notes on namespace management, and subtle differences from `BaseModel`:
123
124 `BaseModel` uses its own `__module__` to find out where it was defined
125 and then looks for symbols to resolve forward references in those globals.
126 On the other hand, `TypeAdapter` can be initialized with arbitrary objects,
127 which may not be types and thus do not have a `__module__` available.
128 So instead we look at the globals in our parent stack frame.
129
130 It is expected that the `ns_resolver` passed to this function will have the correct
131 namespace for the type we're adapting. See the source code for `TypeAdapter.__init__`
132 and `TypeAdapter.rebuild` for various ways to construct this namespace.
133
134 This works for the case where this function is called in a module that
135 has the target of forward references in its scope, but
136 does not always work for more complex cases.
137
138 For example, take the following:
139
140 ```python {title="a.py"}
141 IntList = list[int]
142 OuterDict = dict[str, 'IntList']
143 ```
144
145 ```python {test="skip" title="b.py"}
146 from a import OuterDict
147
148 from pydantic import TypeAdapter
149
150 IntList = int # replaces the symbol the forward reference is looking for
151 v = TypeAdapter(OuterDict)
152 v({'x': 1}) # should fail but doesn't
153 ```
154
155 If `OuterDict` were a `BaseModel`, this would work because it would resolve
156 the forward reference within the `a.py` namespace.
157 But `TypeAdapter(OuterDict)` can't determine what module `OuterDict` came from.
158
159 In other words, the assumption that _all_ forward references exist in the
160 module we are being called from is not technically always true.
161 Although most of the time it is and it works fine for recursive models and such,
162 `BaseModel`'s behavior isn't perfect either and _can_ break in similar ways,
163 so there is no right or wrong between the two.
164
165 But at the very least this behavior is _subtly_ different from `BaseModel`'s.
166 """

196 def __init__(
197 self,
198 type: Any,
199 *,
200 config: ConfigDict | None = None,
201 _parent_depth: int = 2,
202 module: str | None = None,
203 ) -> None:
204 if _type_has_config(type) and config is not None:
205 raise PydanticUserError(
206 'Cannot use `config` when the type is a BaseModel, dataclass or TypedDict.'
207 ' These types can have their own config and setting the config via the `config`'
208 ' parameter to TypeAdapter will not override it, thus the `config` you passed to'
209 ' TypeAdapter becomes meaningless, which is probably not what you want.',
210 code='type-adapter-config-unused',
211 )
212
213 self._type = type
214 self._config = config
215 self._parent_depth = _parent_depth
216 self.pydantic_complete = False
217
218 parent_frame = self._fetch_parent_frame()
219 if isinstance(type, types.FunctionType):
220 # Special case functions, which are *not* pushed to the `NsResolver` stack and without this special case
221 # would only have access to the parent namespace where the `TypeAdapter` was instantiated (if the function is defined
222 # in another module, we need to look at that module's globals).
223 if parent_frame is not None:
224 # `f_locals` is the namespace where the type adapter was instantiated (~ to `f_globals` if at the module level):
225 parent_ns = parent_frame.f_locals
226 else: # pragma: no cover
227 parent_ns = None
228 globalns, localns = _namespace_utils.ns_for_function(
229 type,
230 parent_namespace=parent_ns,
231 )
232 parent_namespace = None
233 else:
234 if parent_frame is not None:
235 globalns = parent_frame.f_globals
236 # Do not provide a local ns if the type adapter happens to be instantiated at the module level:
237 localns = parent_frame.f_locals if parent_frame.f_locals is not globalns else {}
238 else: # pragma: no cover
239 globalns = {}
240 localns = {}
241 parent_namespace = localns
242
243 self._module_name = module or cast(str, globalns.get('__name__', ''))
244 self._init_core_attrs(
245 ns_resolver=_namespace_utils.NsResolver(
246 namespaces_tuple=_namespace_utils.NamespacesTuple(locals=localns, globals=globalns),
247 parent_namespace=parent_namespace,
248 ),
249 force=False,
250 )

252 def _fetch_parent_frame(self) -> FrameType | None:
253 frame = sys._getframe(self._parent_depth)
254 if frame.f_globals.get('__name__') == 'typing':
255 # Because `TypeAdapter` is generic, explicitly parametrizing the class results
256 # in a `typing._GenericAlias` instance, which proxies instantiation calls to the
257 # "real" `TypeAdapter` class and thus adding an extra frame to the call. To avoid
258 # pulling anything from the `typing` module, use the correct frame (the one before):
259 return frame.f_back
260
261 return frame

263 def _init_core_attrs(
264 self, ns_resolver: _namespace_utils.NsResolver, force: bool, raise_errors: bool = False
265 ) -> bool:
266 """Initialize the core schema, validator, and serializer for the type.
267
268 Args:
269 ns_resolver: The namespace resolver to use when building the core schema for the adapted type.
270 force: Whether to force the construction of the core schema, validator, and serializer.
271 If `force` is set to `False` and `_defer_build` is `True`, the core schema, validator, and serializer will be set to mocks.
272 raise_errors: Whether to raise errors if initializing any of the core attrs fails.
273
274 Returns:
275 `True` if the core schema, validator, and serializer were successfully initialized, otherwise `False`.
276
277 Raises:
278 PydanticUndefinedAnnotation: If `PydanticUndefinedAnnotation` occurs in`__get_pydantic_core_schema__`
279 and `raise_errors=True`.
280 """
281 if not force and self._defer_build:
282 _mock_val_ser.set_type_adapter_mocks(self)
283 self.pydantic_complete = False
284 return False
285
286 try:
287 self.core_schema = _getattr_no_parents(self._type, '__pydantic_core_schema__')
288 self.validator = _getattr_no_parents(self._type, '__pydantic_validator__')
289 self.serializer = _getattr_no_parents(self._type, '__pydantic_serializer__')
290
291 # TODO: we don't go through the rebuild logic here directly because we don't want
292 # to repeat all of the namespace fetching logic that we've already done
293 # so we simply skip to the block below that does the actual schema generation
294 if (
295 isinstance(self.core_schema, _mock_val_ser.MockCoreSchema)
296 or isinstance(self.validator, _mock_val_ser.MockValSer)
297 or isinstance(self.serializer, _mock_val_ser.MockValSer)
298 ):
299 raise AttributeError()
300 except AttributeError:
301 config_wrapper = _config.ConfigWrapper(self._config)
302
303 schema_generator = _generate_schema.GenerateSchema(config_wrapper, ns_resolver=ns_resolver)
304
305 try:
306 core_schema = schema_generator.generate_schema(self._type)
307 except PydanticUndefinedAnnotation:
308 if raise_errors:
309 raise
310 _mock_val_ser.set_type_adapter_mocks(self)
311 return False
312
313 try:
314 self.core_schema = schema_generator.clean_schema(core_schema)
315 except _generate_schema.InvalidSchemaError:
316 _mock_val_ser.set_type_adapter_mocks(self)
317 return False
318
319 core_config = config_wrapper.core_config(None)
320
321 self.validator = create_schema_validator(
322 schema=self.core_schema,
323 schema_type=self._type,
324 schema_type_module=self._module_name,
325 schema_type_name=str(self._type),
326 schema_kind='TypeAdapter',
327 config=core_config,
328 plugin_settings=config_wrapper.plugin_settings,
329 )
330 self.serializer = SchemaSerializer(self.core_schema, core_config)
331
332 self.pydantic_complete = True
333 return True

352 def rebuild(
353 self,
354 *,
355 force: bool = False,
356 raise_errors: bool = True,
357 _parent_namespace_depth: int = 2,
358 _types_namespace: _namespace_utils.MappingNamespace | None = None,
359 ) -> bool | None:
360 """Try to rebuild the pydantic-core schema for the adapter's type.
361
362 This may be necessary when one of the annotations is a ForwardRef which could not be resolved during
363 the initial attempt to build the schema, and automatic rebuilding fails.
364
365 Args:
366 force: Whether to force the rebuilding of the type adapter's schema, defaults to `False`.
367 raise_errors: Whether to raise errors, defaults to `True`.
368 _parent_namespace_depth: Depth at which to search for the [parent frame][frame-objects]. This
369 frame is used when resolving forward annotations during schema rebuilding, by looking for
370 the locals of this frame. Defaults to 2, which will result in the frame where the method
371 was called.
372 _types_namespace: An explicit types namespace to use, instead of using the local namespace
373 from the parent frame. Defaults to `None`.
374
375 Returns:
376 Returns `None` if the schema is already "complete" and rebuilding was not required.
377 If rebuilding _was_ required, returns `True` if rebuilding was successful, otherwise `False`.
378 """
379 if not force and self.pydantic_complete:
380 return None
381
382 if _types_namespace is not None:
383 rebuild_ns = _types_namespace
384 elif _parent_namespace_depth > 0:
385 rebuild_ns = _typing_extra.parent_frame_namespace(parent_depth=_parent_namespace_depth, force=True) or {}
386 else:
387 rebuild_ns = {}
388
389 # we have to manually fetch globals here because there's no type on the stack of the NsResolver
390 # and so we skip the globalns = get_module_ns_of(typ) call that would normally happen
391 globalns = sys._getframe(max(_parent_namespace_depth - 1, 1)).f_globals
392 ns_resolver = _namespace_utils.NsResolver(
393 namespaces_tuple=_namespace_utils.NamespacesTuple(locals=rebuild_ns, globals=globalns),
394 parent_namespace=rebuild_ns,
395 )
396 return self._init_core_attrs(ns_resolver=ns_resolver, force=True, raise_errors=raise_errors)
