220class FreePortFactory:
221 """
222 Manages port generation based on specified socket kind, ensuring no duplicate
223 ports are generated.
224
225 This class provides functionality for generating available free ports on the
226 system. It is initialized with a specific socket kind and can generate ports
227 for given address families while avoiding reuse of previously generated ports.
228
229 Users should not instantiate this class directly, but use the
230 ``free_tcp_port_factory`` and ``free_udp_port_factory`` fixtures instead. For simple
231 uses cases, ``free_tcp_port`` and ``free_udp_port`` can be used instead.
232 """
233
234 def __init__(self, kind: socket.SocketKind) -> None:
235 self._kind = kind
236 self._generated = set[int]()
237
238 @property
239 def kind(self) -> socket.SocketKind:
240 """
241 The type of socket connection (e.g., :data:`~socket.SOCK_STREAM` or
242 :data:`~socket.SOCK_DGRAM`) used to bind for checking port availability
243
244 """
245 return self._kind
246
247 def __call__(self, family: socket.AddressFamily | None = None) -> int:
248 """
249 Return an unbound port for the given address family.
250
251 :param family: if omitted, both IPv4 and IPv6 addresses will be tried
252 :return: a port number
253
254 """
255 if family is not None:
256 families = [family]
257 else:
258 families = [socket.AF_INET]
259 if socket.has_ipv6:
260 families.append(socket.AF_INET6)
261
262 while True:
263 port = 0
264 with ExitStack() as stack:
265 for family in families:
266 sock = stack.enter_context(socket.socket(family, self._kind))
267 addr = "::1" if family == socket.AF_INET6 else "127.0.0.1"
268 try:
269 sock.bind((addr, port))
270 except OSError:
271 break
272
273 if not port:
274 port = sock.getsockname()[1]
275 else:
276 if port not in self._generated:
277 self._generated.add(port)
278 return port
