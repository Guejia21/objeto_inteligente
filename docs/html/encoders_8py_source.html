38def decimal_encoder(dec_value: Decimal) -> Union[int, float]:
39 """
40 Encodes a Decimal as int of there's no exponent, otherwise float
41
42 This is useful when we use ConstrainedDecimal to represent Numeric(x,0)
43 where a integer (but not int typed) is used. Encoding this as a float
44 results in failed round-tripping between encode and parse.
45 Our Id type is a prime example of this.
46
47 >>> decimal_encoder(Decimal("1.0"))
48 1.0
49
50 >>> decimal_encoder(Decimal("1"))
51 1
52 """
53 if dec_value.as_tuple().exponent >= 0: # type: ignore[operator]
54 return int(dec_value)
55 else:
56 return float(dec_value)
