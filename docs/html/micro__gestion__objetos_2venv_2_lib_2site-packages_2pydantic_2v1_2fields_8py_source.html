197 def update_from_config(self, from_config: Dict[str, Any]) -> None:
198 """
199 Update this FieldInfo based on a dict from get_field_info, only fields which have not been set are dated.
200 """
201 for attr_name, value in from_config.items():
202 try:
203 current_value = getattr(self, attr_name)
204 except AttributeError:
205 # attr_name is not an attribute of FieldInfo, it should therefore be added to extra
206 # (except if extra already has this value!)
207 self.extra.setdefault(attr_name, value)
208 else:
209 if current_value is self.__field_constraints__.get(attr_name, None):
210 setattr(self, attr_name, value)
211 elif attr_name == 'exclude':
212 self.exclude = ValueItems.merge(value, current_value)
213 elif attr_name == 'include':
214 self.include = ValueItems.merge(value, current_value, intersect=True)

439 @staticmethod
440 def _get_field_info(
441 field_name: str, annotation: Any, value: Any, config: Type['BaseConfig']
442 ) -> Tuple[FieldInfo, Any]:
443 """
444 Get a FieldInfo from a root typing.Annotated annotation, value, or config default.
445
446 The FieldInfo may be set in typing.Annotated or the value, but not both. If neither contain
447 a FieldInfo, a new one will be created using the config.
448
449 :param field_name: name of the field for use in error messages
450 :param annotation: a type hint such as `str` or `Annotated[str, Field(..., min_length=5)]`
451 :param value: the field's assigned value
452 :param config: the model's config object
453 :return: the FieldInfo contained in the `annotation`, the value, or a new one from the config.
454 """
455 field_info_from_config = config.get_field_info(field_name)
456
457 field_info = None
458 if get_origin(annotation) is Annotated:
459 field_infos = [arg for arg in get_args(annotation)[1:] if isinstance(arg, FieldInfo)]
460 if len(field_infos) > 1:
461 raise ValueError(f'cannot specify multiple `Annotated` `Field`s for {field_name!r}')
462 field_info = next(iter(field_infos), None)
463 if field_info is not None:
464 field_info = copy.copy(field_info)
465 field_info.update_from_config(field_info_from_config)
466 if field_info.default not in (Undefined, Required):
467 raise ValueError(f'`Field` default cannot be set in `Annotated` for {field_name!r}')
468 if value is not Undefined and value is not Required:
469 # check also `Required` because of `validate_arguments` that sets `...` as default value
470 field_info.default = value
471
472 if isinstance(value, FieldInfo):
473 if field_info is not None:
474 raise ValueError(f'cannot specify `Annotated` and value `Field`s together for {field_name!r}')
475 field_info = value
476 field_info.update_from_config(field_info_from_config)
477 elif field_info is None:
478 field_info = FieldInfo(value, **field_info_from_config)
479 value = None if field_info.default_factory is not None else field_info.default
480 field_info._validate()
481 return field_info, value

537 def prepare(self) -> None:
538 """
539 Prepare the field but inspecting self.default, self.type_ etc.
540
541 Note: this method is **not** idempotent (because _type_analysis is not idempotent),
542 e.g. calling it it multiple times may modify the field and configure it incorrectly.
543 """
544 self._set_default_and_type()
545 if self.type_.__class__ is ForwardRef or self.type_.__class__ is DeferredType:
546 # self.type_ is currently a ForwardRef and there's nothing we can do now,
547 # user will need to call model.update_forward_refs()
548 return

550 self._type_analysis()
551 if self.required is Undefined:
552 self.required = True
553 if self.default is Undefined and self.default_factory is None:
554 self.default = None

557 def _set_default_and_type(self) -> None:
558 """
559 Set the default value, infer the type if needed and check if `None` value is valid.
560 """
561 if self.default_factory is not None:
562 if self.type_ is Undefined:
563 raise errors_.ConfigError(
564 f'you need to set the type of field {self.name!r} when using `default_factory`'
565 )
566 return
567
568 default_value = self.get_default()
569
570 if default_value is not None and self.type_ is Undefined:
571 self.type_ = default_value.__class__
572 self.outer_type_ = self.type_
573 self.annotation = self.type_
574
575 if self.type_ is Undefined:
576 raise errors_.ConfigError(f'unable to infer type for attribute "{self.name}"')
577
578 if self.required is False and default_value is None:
579 self.allow_none = True

581 def _type_analysis(self) -> None: # noqa: C901 (ignore complexity)
582 # typing interface is horrible, we have to do some ugly checks
583 if lenient_issubclass(self.type_, JsonWrapper):
584 self.type_ = self.type_.inner_type
585 self.parse_json = True
586 elif lenient_issubclass(self.type_, Json):
587 self.type_ = Any
588 self.parse_json = True
589 elif isinstance(self.type_, TypeVar):
590 if self.type_.__bound__:
591 self.type_ = self.type_.__bound__
592 elif self.type_.__constraints__:
593 self.type_ = Union[self.type_.__constraints__]
594 else:
595 self.type_ = Any
596 elif is_new_type(self.type_):
597 self.type_ = new_type_supertype(self.type_)
598
599 if self.type_ is Any or self.type_ is object:
600 if self.required is Undefined:
601 self.required = False
602 self.allow_none = True
603 return
604 elif self.type_ is Pattern or self.type_ is re.Pattern:
605 # python 3.7 only, Pattern is a typing object but without sub fields
606 return
607 elif is_literal_type(self.type_):
608 return
609 elif is_typeddict(self.type_):
610 return
611
612 if is_finalvar(self.type_):
613 self.final = True
614
615 if self.type_ is Final:
616 self.type_ = Any
617 else:
618 self.type_ = get_args(self.type_)[0]
619
620 self._type_analysis()
621 return
622
623 origin = get_origin(self.type_)
624
625 if origin is Annotated or is_typeddict_special(origin):
626 self.type_ = get_args(self.type_)[0]
627 self._type_analysis()
628 return
629
630 if self.discriminator_key is not None and not is_union(origin):
631 raise TypeError('`discriminator` can only be used with `Union` type with more than one variant')
632
633 # add extra check for `collections.abc.Hashable` for python 3.10+ where origin is not `None`
634 if origin is None or origin is CollectionsHashable:
635 # field is not "typing" object eg. Union, Dict, List etc.
636 # allow None for virtual superclasses of NoneType, e.g. Hashable
637 if isinstance(self.type_, type) and isinstance(None, self.type_):
638 self.allow_none = True
639 return
640 elif origin is Callable:
641 return
642 elif is_union(origin):
643 types_ = []
644 for type_ in get_args(self.type_):
645 if is_none_type(type_) or type_ is Any or type_ is object:
646 if self.required is Undefined:
647 self.required = False
648 self.allow_none = True
649 if is_none_type(type_):
650 continue
651 types_.append(type_)
652
653 if len(types_) == 1:
654 # Optional[]
655 self.type_ = types_[0]
656 # this is the one case where the "outer type" isn't just the original type
657 self.outer_type_ = self.type_
658 # re-run to correctly interpret the new self.type_
659 self._type_analysis()
660 else:
661 self.sub_fields = [self._create_sub_type(t, f'{self.name}_{display_as_type(t)}') for t in types_]
662
663 if self.discriminator_key is not None:
664 self.prepare_discriminated_union_sub_fields()
