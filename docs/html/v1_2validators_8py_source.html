481def make_literal_validator(type_: Any) -> Callable[[Any], Any]:
482 permitted_choices = all_literal_values(type_)
483
484 # To have a O(1) complexity and still return one of the values set inside the `Literal`,
485 # we create a dict with the set values (a set causes some problems with the way intersection works).
486 # In some cases the set value and checked value can indeed be different (see `test_literal_validator_str_enum`)
487 allowed_choices = {v: v for v in permitted_choices}
488
489 def literal_validator(v: Any) -> Any:
490 try:
491 return allowed_choices[v]
492 except (KeyError, TypeError):
493 raise errors.WrongConstantError(given=v, permitted=permitted_choices)
494
495 return literal_validator

598 from pydantic.v1.annotated_types import create_model_from_namedtuple

608 annotations = NamedTupleModel.__annotations__

699def find_validators( # noqa: C901 (ignore complexity)
700 type_: Type[Any], config: Type['BaseConfig']
701) -> Generator[AnyCallable, None, None]:
702 from pydantic.v1.dataclasses import is_builtin_dataclass, make_dataclass_validator
703
704 if type_ is Any or type_ is object:
705 return
706 type_type = type_.__class__
707 if type_type == ForwardRef or type_type == TypeVar:
708 return
709
710 if is_none_type(type_):
711 yield none_validator
712 return
713 if type_ is Pattern or type_ is re.Pattern:
714 yield pattern_validator
715 return
716 if type_ is Hashable or type_ is CollectionsHashable:
717 yield hashable_validator
718 return
719 if is_callable_type(type_):
720 yield callable_validator
721 return
722 if is_literal_type(type_):
723 yield make_literal_validator(type_)
724 return
725 if is_builtin_dataclass(type_):
726 yield from make_dataclass_validator(type_, config)
727 return
728 if type_ is Enum:
729 yield enum_validator
730 return
731 if type_ is IntEnum:
732 yield int_enum_validator
733 return
734 if is_namedtuple(type_):
735 yield tuple_validator
736 yield make_namedtuple_validator(type_, config)
737 return
738 if is_typeddict(type_):
739 yield make_typeddict_validator(type_, config)
740 return
741
742 class_ = get_class(type_)
743 if class_ is not None:
744 if class_ is not Any and isinstance(class_, type):
745 yield make_class_validator(class_)
746 else:
747 yield any_class_validator
748 return
749
750 for val_type, validators in _VALIDATORS:
751 try:
752 if issubclass(type_, val_type):
753 for v in validators:
754 if isinstance(v, IfConfig):
755 if v.check(config):
756 yield v.validator
757 else:
758 yield v
759 return
760 except TypeError:
761 raise RuntimeError(f'error checking inheritance of {type_!r} (type: {display_as_type(type_)})')
762
763 if config.arbitrary_types_allowed:
764 yield make_arbitrary_type_validator(type_)
765 else:
766 if hasattr(type_, '__pydantic_core_schema__'):
767 warn(f'Mixing V1 and V2 models is not supported. `{type_.__name__}` is a V2 model.', UserWarning)
768 raise RuntimeError(f'no validator found for {type_}, see `arbitrary_types_allowed` in Config')
