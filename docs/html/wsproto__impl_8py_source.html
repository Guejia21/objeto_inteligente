165 def handle_connect(self, event: events.Request) -> None:
166 headers = [(b"host", event.host.encode())]
167 headers += [(key.lower(), value) for key, value in event.extra_headers]
168 raw_path, _, query_string = event.target.partition("?")
169 path = unquote(raw_path)
170 full_path = self.root_path + path
171 full_raw_path = self.root_path.encode("ascii") + raw_path.encode("ascii")
172 self.scope: WebSocketScope = {
173 "type": "websocket",
174 "asgi": {"version": self.config.asgi_version, "spec_version": "2.4"},
175 "http_version": "1.1",
176 "scheme": self.scheme,
177 "server": self.server,
178 "client": self.client,
179 "root_path": self.root_path,
180 "path": full_path,
181 "raw_path": full_raw_path,
182 "query_string": query_string.encode("ascii"),
183 "headers": headers,
184 "subprotocols": event.subprotocols,
185 "state": self.app_state.copy(),
186 "extensions": {"websocket.http.response": {}},
187 }
188 self.queue.put_nowait({"type": "websocket.connect"})
189 task = self.loop.create_task(self.run_asgi())
190 task.add_done_callback(self.on_task_complete)
191 self.tasks.add(task)

221 def send_500_response(self) -> None:
222 if self.response_started or self.handshake_complete:
223 return # we cannot send responses anymore

233 async def run_asgi(self) -> None:
234 try:
235 result = await self.app(self.scope, self.receive, self.send) # type: ignore[func-returns-value]
236 except ClientDisconnected:

238 except BaseException:

251 async def send(self, message: ASGISendEvent) -> None:
252 await self.writable.wait()
253
254 message_type = message["type"]
255
256 if not self.handshake_complete:
257 if message_type == "websocket.accept":
258 message = cast(WebSocketAcceptEvent, message)
259 self.logger.info(
260 '%s - "WebSocket %s" [accepted]',
261 get_client_addr(self.scope),
262 get_path_with_query_string(self.scope),
263 )
264 subprotocol = message.get("subprotocol")
265 extra_headers = self.default_headers + list(message.get("headers", []))
266 extensions: list[Extension] = []
267 if self.config.ws_per_message_deflate:
268 extensions.append(PerMessageDeflate())
269 if not self.transport.is_closing():
270 self.handshake_complete = True
271 output = self.conn.send(
272 wsproto.events.AcceptConnection(
273 subprotocol=subprotocol,
274 extensions=extensions,
275 extra_headers=extra_headers,
276 )
277 )
278 self.transport.write(output)
279
280 elif message_type == "websocket.close":
281 self.queue.put_nowait({"type": "websocket.disconnect", "code": 1006})
282 self.logger.info(
283 '%s - "WebSocket %s" 403',
284 get_client_addr(self.scope),
285 get_path_with_query_string(self.scope),
286 )
287 self.handshake_complete = True
288 self.close_sent = True
289 event = events.RejectConnection(status_code=403, headers=[])
290 output = self.conn.send(event)
291 self.transport.write(output)
292 self.transport.close()
293
294 elif message_type == "websocket.http.response.start":
295 message = cast(WebSocketResponseStartEvent, message)
296 # ensure status code is in the valid range
297 if not (100 <= message["status"] < 600):
298 msg = "Invalid HTTP status code '%d' in response."
299 raise RuntimeError(msg % message["status"])
300 self.logger.info(
301 '%s - "WebSocket %s" %d',
302 get_client_addr(self.scope),
303 get_path_with_query_string(self.scope),
304 message["status"],
305 )
306 self.handshake_complete = True
307 event = events.RejectConnection(
308 status_code=message["status"],
309 headers=list(message["headers"]),
310 has_body=True,
311 )
312 output = self.conn.send(event)
313 self.transport.write(output)
314 self.response_started = True
315
316 else:
317 msg = (
318 "Expected ASGI message 'websocket.accept', 'websocket.close' "
319 "or 'websocket.http.response.start' "
320 "but got '%s'."
321 )
322 raise RuntimeError(msg % message_type)
323
324 elif not self.close_sent and not self.response_started:
325 try:
326 if message_type == "websocket.send":
327 message = cast(WebSocketSendEvent, message)
328 bytes_data = message.get("bytes")
329 text_data = message.get("text")
330 data = text_data if bytes_data is None else bytes_data
331 output = self.conn.send(wsproto.events.Message(data=data)) # type: ignore
332 if not self.transport.is_closing():
333 self.transport.write(output)
334
335 elif message_type == "websocket.close":
336 message = cast(WebSocketCloseEvent, message)
337 self.close_sent = True
338 code = message.get("code", 1000)
339 reason = message.get("reason", "") or ""
340 self.queue.put_nowait({"type": "websocket.disconnect", "code": code, "reason": reason})
341 output = self.conn.send(wsproto.events.CloseConnection(code=code, reason=reason))
342 if not self.transport.is_closing():
343 self.transport.write(output)
344 self.transport.close()
345
346 else:
347 msg = "Expected ASGI message 'websocket.send' or 'websocket.close', but got '%s'."
348 raise RuntimeError(msg % message_type)
349 except LocalProtocolError as exc:
350 raise ClientDisconnected from exc
351 elif self.response_started:
352 if message_type == "websocket.http.response.body":
353 message = cast("WebSocketResponseBodyEvent", message)
354 body_finished = not message.get("more_body", False)
355 reject_data = events.RejectData(data=message["body"], body_finished=body_finished)
356 output = self.conn.send(reject_data)
357 self.transport.write(output)
358
359 if body_finished:
360 self.queue.put_nowait({"type": "websocket.disconnect", "code": 1006})
361 self.close_sent = True
362 self.transport.close()
363
364 else:
365 msg = "Expected ASGI message 'websocket.http.response.body' but got '%s'."
366 raise RuntimeError(msg % message_type)
367
368 else:
369 msg = "Unexpected ASGI message '%s', after sending 'websocket.close'."
370 raise RuntimeError(msg % message_type)
