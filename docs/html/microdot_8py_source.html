64def urldecode(s):
65 if isinstance(s, str):
66 s = s.encode()
67 s = s.replace(b'+', b' ')
68 parts = s.split(b'%')
69 if len(parts) == 1:
70 return s.decode()
71 result = [parts[0]]
72 for item in parts[1:]:
73 if item == b'':
74 result.append(b'%')
75 else:
76 code = item[:2]
77 result.append(bytes([int(code, 16)]))
78 result.append(item[2:])
79 return b''.join(result).decode()

88class NoCaseDict(dict):
89 """A subclass of dictionary that holds case-insensitive keys.
90
91 :param initial_dict: an initial dictionary of key/value pairs to
92 initialize this object with.
93
94 Example::
95
96 >>> d = NoCaseDict()
97 >>> d['Content-Type'] = 'text/html'
98 >>> print(d['Content-Type'])
99 text/html
100 >>> print(d['content-type'])
101 text/html
102 >>> print(d['CONTENT-TYPE'])
103 text/html
104 >>> del d['cOnTeNt-TyPe']
105 >>> print(d)
106 {}
107 """
108 def __init__(self, initial_dict=None):
109 super().__init__(initial_dict or {})
110 self.keymap = {k.lower(): k for k in self.keys() if k.lower() != k}
111
112 def __setitem__(self, key, value):
113 kl = key.lower()
114 key = self.keymap.get(kl, key)
115 if kl != key:
116 self.keymap[kl] = key
117 super().__setitem__(key, value)
118
119 def __getitem__(self, key):
120 kl = key.lower()
121 return super().__getitem__(self.keymap.get(kl, kl))
122
123 def __delitem__(self, key):
124 kl = key.lower()
125 super().__delitem__(self.keymap.get(kl, kl))
126
127 def __contains__(self, key):
128 kl = key.lower()
129 return self.keymap.get(kl, kl) in self.keys()

131 def get(self, key, default=None):
132 kl = key.lower()
133 return super().get(self.keymap.get(kl, kl), default)

140def mro(cls): # pragma: no cover
141 """Return the method resolution order of a class.
142
143 This is a helper function that returns the method resolution order of a
144 class. It is used by Microdot to find the best error handler to invoke for
145 the raised exception.
146
147 In CPython, this function returns the ``__mro__`` attribute of the class.
148 In MicroPython, this function implements a recursive depth-first scanning
149 of the class hierarchy.
150 """
151 if hasattr(cls, 'mro'):
152 return cls.__mro__
153
154 def _mro(cls):
155 m = [cls]
156 for base in cls.__bases__:
157 m += _mro(base)
158 return m
159
160 mro_list = _mro(cls)
161
162 # If a class appears multiple times (due to multiple inheritance) remove
163 # all but the last occurence. This matches the method resolution order
164 # of MicroPython, but not CPython.
165 mro_pruned = []
166 for i in range(len(mro_list)):
167 base = mro_list.pop(0)
168 if base not in mro_list:
169 mro_pruned.append(base)
170 return mro_pruned

173class MultiDict(dict):
174 """A subclass of dictionary that can hold multiple values for the same
175 key. It is used to hold key/value pairs decoded from query strings and
176 form submissions.
177
178 :param initial_dict: an initial dictionary of key/value pairs to
179 initialize this object with.
180
181 Example::
182
183 >>> d = MultiDict()
184 >>> d['sort'] = 'name'
185 >>> d['sort'] = 'email'
186 >>> print(d['sort'])
187 'name'
188 >>> print(d.getlist('sort'))
189 ['name', 'email']
190 """
191 def __init__(self, initial_dict=None):
192 super().__init__()
193 if initial_dict:
194 for key, value in initial_dict.items():
195 self[key] = value
196
197 def __setitem__(self, key, value):
198 if key not in self:
199 super().__setitem__(key, [])
200 super().__getitem__(key).append(value)
201
202 def __getitem__(self, key):
203 return super().__getitem__(key)[0]

205 def get(self, key, default=None, type=None):
206 """Return the value for a given key.
207
208 :param key: The key to retrieve.
209 :param default: A default value to use if the key does not exist.
210 :param type: A type conversion callable to apply to the value.
211
212 If the multidict contains more than one value for the requested key,
213 this method returns the first value only.
214
215 Example::
216
217 >>> d = MultiDict()
218 >>> d['age'] = '42'
219 >>> d.get('age')
220 '42'
221 >>> d.get('age', type=int)
222 42
223 >>> d.get('name', default='noname')
224 'noname'
225 """
226 if key not in self:
227 return default
228 value = self[key]
229 if type is not None:
230 value = type(value)
231 return value

234 """Return all the values for a given key.
235
236 :param key: The key to retrieve.
237 :param type: A type conversion callable to apply to the values.
238
239 If the requested key does not exist in the dictionary, this method

382 async def create(app, client_reader, client_writer, client_addr):
383 """Create a request object.
384
385 :param app: The Microdot application instance.
386 :param client_reader: An input stream from where the request data can
387 be read.
388 :param client_writer: An output stream where the response data can be
389 written.
390 :param client_addr: The address of the client, as a tuple.
391
392 This method is a coroutine. It returns a newly created ``Request``
393 object.
394 """
395 # request line
396 line = (await Request._safe_readline(client_reader)).strip().decode()
397 if not line: # pragma: no cover
398 return None
399 method, url, http_version = line.split()
400 http_version = http_version.split('/', 1)[1]
401
402 # headers
403 headers = NoCaseDict()
404 content_length = 0
405 while True:
406 line = (await Request._safe_readline(
407 client_reader)).strip().decode()
408 if line == '':
409 break
410 header, value = line.split(':', 1)
411 value = value.strip()
412 headers[header] = value
413 if header.lower() == 'content-length':
414 content_length = int(value)
415
416 # body
417 body = b''
418 if content_length and content_length <= Request.max_body_length:
419 body = await client_reader.readexactly(content_length)
420 stream = None
421 else:
422 body = b''
423 stream = client_reader
424
425 return Request(app, client_addr, method, url, http_version, headers,
426 body=body, stream=stream,
427 sock=(client_reader, client_writer))

762 @classmethod
763 def send_file(cls, filename, status_code=200, content_type=None,
764 stream=None, max_age=None, compressed=False,
765 file_extension=''):
766 """Send file contents in a response.
767
768 :param filename: The filename of the file.
769 :param status_code: The 3xx status code to use for the redirect. The
770 default is 302.
771 :param content_type: The ``Content-Type`` header to use in the
772 response. If omitted, it is generated
773 automatically from the file extension of the
774 ``filename`` parameter.
775 :param stream: A file-like object to read the file contents from. If
776 a stream is given, the ``filename`` parameter is only
777 used when generating the ``Content-Type`` header.
778 :param max_age: The ``Cache-Control`` header's ``max-age`` value in
779 seconds. If omitted, the value of the
780 :attr:`Response.default_send_file_max_age` attribute is
781 used.
782 :param compressed: Whether the file is compressed. If ``True``, the
783 ``Content-Encoding`` header is set to ``gzip``. A
784 string with the header value can also be passed.
785 Note that when using this option the file must have
786 been compressed beforehand. This option only sets
787 the header.
788 :param file_extension: A file extension to append to the ``filename``
789 parameter when opening the file, including the
790 dot. The extension given here is not considered
791 when generating the ``Content-Type`` header.
792

794 filenames provided by the user without validating and sanitizing them
795 first.

799 ext = filename[:-3].split('.')[-1]
800 else:
801 ext = filename.split('.')[-1]

803 content_type = Response.types_map[ext]
804 else:
805 content_type = 'application/octet-stream'
806 headers = {'Content-Type': content_type}
807
808 if max_age is None:
809 max_age = cls.default_send_file_max_age
810 if max_age is not None:
811 headers['Cache-Control'] = 'max-age={}'.format(max_age)
812
813 if compressed:
814 headers['Content-Encoding'] = compressed \
815 if isinstance(compressed, str) else 'gzip'
816

818 return cls(body=f, status_code=status_code, headers=headers)

821class URLPattern():
822 """A class that represents the URL pattern for a route.
823
824 :param url_pattern: The route URL pattern, which can include static and
825 dynamic path segments. Dynamic segments are enclosed in
826 ``<`` and ``>``. The type of the segment can be given
827 as a prefix, separated from the name with a colon.
828 Supported types are ``string`` (the default),
829 ``int`` and ``path``. Custom types can be registered
830 using the :meth:`URLPattern.register_type` method.
831 """
832
833 segment_patterns = {
834 'string': '/([^/]+)',
835 'int': '/(-?\\d+)',
836 'path': '/(.+)',
837 }
838 segment_parsers = {
839 'int': lambda value: int(value),
840 }
841
842 @classmethod
843 def register_type(cls, type_name, pattern='[^/]+', parser=None):
844 """Register a new URL segment type.
845
846 :param type_name: The name of the segment type to register.
847 :param pattern: The regular expression pattern to use when matching
848 this segment type. If not given, a default matcher for
849 a single path segment is used.
850 :param parser: A callable that will be used to parse and transform the
851 value of the segment. If omitted, the value is returned
852 as a string.
853 """
854 cls.segment_patterns[type_name] = '/({})'.format(pattern)
855 cls.segment_parsers[type_name] = parser
856
857 def __init__(self, url_pattern):
858 self.url_pattern = url_pattern
859 self.segments = []
860 self.regex = None
861
862 def compile(self):
863 """Generate a regular expression for the URL pattern.
864
865 This method is automatically invoked the first time the URL pattern is
866 matched against a path.
867 """
868 pattern = ''
869 for segment in self.url_pattern.lstrip('/').split('/'):
870 if segment and segment[0] == '<':
871 if segment[-1] != '>':
872 raise ValueError('invalid URL pattern')
873 segment = segment[1:-1]
874 if ':' in segment:
875 type_, name = segment.rsplit(':', 1)
876 else:
877 type_ = 'string'
878 name = segment
879 parser = None
880 if type_.startswith('re:'):
881 pattern += '/({pattern})'.format(pattern=type_[3:])
882 else:
883 if type_ not in self.segment_patterns:
884 raise ValueError('invalid URL segment type')
885 pattern += self.segment_patterns[type_]
886 parser = self.segment_parsers.get(type_)
887 self.segments.append({'parser': parser, 'name': name,
888 'type': type_})
889 else:
890 pattern += '/' + segment
891 self.segments.append({'parser': None})
892 self.regex = re.compile('^' + pattern + '$')
893 return self.regex
894
895 def match(self, path):
896 """Match a path against the URL pattern.
897
898 Returns a dictionary with the values of all dynamic path segments if a
899 matche is found, or ``None`` if the path does not match this pattern.
900 """
901 args = {}
902 g = (self.regex or self.compile()).match(path)
903 if not g:
904 return
905 i = 1
906 for segment in self.segments:
907 if 'name' not in segment:
908 continue
909 arg = g.group(i)
910 if segment['parser']:
911 arg = self.segment_parsers[segment['type']](arg)
912 if arg is None:
913 return
914 args[segment['name']] = arg
915 i += 1
916 return args
917
918 def __repr__(self): # pragma: no cover
919 return 'URLPattern: {}'.format(self.url_pattern)

1337 def find_route(self, req):
1338 method = req.method.upper()
1339 if method == 'OPTIONS' and self.options_handler:
1340 return self.options_handler(req), '', None
1341 if method == 'HEAD':
1342 method = 'GET'

1349 if req.url_args is not None:
1350 p = url_prefix
1351 s = subapp
1352 if method in route_methods:
1353 f = route_handler
1354 break
1355 else:
1356 f = 405
1357 return f, p, s

1369 async def handle_request(self, reader, writer):
1370 req = None
1371 try:
1372 req = await Request.create(self, reader, writer,
1373 writer.get_extra_info('peername'))
1374 except Exception as exc: # pragma: no cover
1375 print_exception(exc)

1377 res = await self.dispatch_request(req)
1378 try:
1379 if res != Response.already_handled: # pragma: no branch
1380 await res.write(writer)
1381 await writer.aclose()
1382 except OSError as exc: # pragma: no cover
1383 if exc.errno in MUTED_SOCKET_ERRORS:
1384 pass
1385 else:
1386 raise
1387 if self.debug and req: # pragma: no cover
1388 print('{method} {path} {status_code}'.format(
1389 method=req.method, path=req.path,
1390 status_code=res.status_code))

1395 if req and req.subapp else []

1399 async def error_response(self, req, status_code, reason=None):
1400 if req and req.subapp and status_code in req.subapp.error_handlers:
1401 return await invoke_handler(
1402 req.subapp.error_handlers[status_code], req)
1403 elif status_code in self.error_handlers:
1404 return await invoke_handler(self.error_handlers[status_code], req)
1405 return reason or 'N/A', status_code

1407 async def dispatch_request(self, req):
1408 after_request_handled = False
1409 if req:
1410 if req.content_length > req.max_content_length:
1411 # the request body is larger than allowed
1412 res = await self.error_response(req, 413, 'Payload too large')
1413 else:
1414 # find the route in the app's URL map
1415 f, req.url_prefix, req.subapp = self.find_route(req)
1416
1417 try:
1418 res = None
1419 if callable(f):
1420 # invoke the before request handlers
1421 for handler in self.get_request_handlers(
1422 req, 'before_request', False):
1423 res = await invoke_handler(handler, req)
1424 if res:
1425 break
1426
1427 # invoke the endpoint handler
1428 if res is None:
1429 res = await invoke_handler(f, req, **req.url_args)
1430
1431 # process the response
1432 if isinstance(res, int):
1433 # an integer response is taken as a status code
1434 # with an empty body
1435 res = '', res
1436 if isinstance(res, tuple):
1437 # handle a tuple response
1438 if isinstance(res[0], int):
1439 # a tuple that starts with an int has an empty
1440 # body
1441 res = ('', res[0],
1442 res[1] if len(res) > 1 else {})
1443 body = res[0]
1444 if isinstance(res[1], int):
1445 # extract the status code and headers (if
1446 # available)
1447 status_code = res[1]
1448 headers = res[2] if len(res) > 2 else {}
1449 else:
1450 # if the status code is missing, assume 200
1451 status_code = 200
1452 headers = res[1]
1453 res = Response(body, status_code, headers)
1454 elif not isinstance(res, Response):
1455 # any other response types are wrapped in a
1456 # Response object
1457 res = Response(res)
1458
1459 # invoke the after request handlers
1460 for handler in self.get_request_handlers(
1461 req, 'after_request', True):
1462 res = await invoke_handler(
1463 handler, req, res) or res
1464 for handler in req.after_request_handlers:
1465 res = await invoke_handler(
1466 handler, req, res) or res
1467 after_request_handled = True
1468 elif isinstance(f, dict):
1469 # the response from an OPTIONS request is a dict with
1470 # headers
1471 res = Response(headers=f)
1472 else:
1473 # if the route is not found, return a 404 or 405
1474 # response as appropriate
1475 res = await self.error_response(req, f, 'Not found')
1476 except HTTPException as exc:
1477 # an HTTP exception was raised while handling this request
