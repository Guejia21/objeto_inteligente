187def get_flat_dependant(
188 dependant: Dependant,
189 *,
190 skip_repeats: bool = False,
191 visited: Optional[List[CacheKey]] = None,
192) -> Dependant:
193 if visited is None:
194 visited = []
195 visited.append(dependant.cache_key)
196
197 flat_dependant = Dependant(
198 path_params=dependant.path_params.copy(),
199 query_params=dependant.query_params.copy(),
200 header_params=dependant.header_params.copy(),
201 cookie_params=dependant.cookie_params.copy(),
202 body_params=dependant.body_params.copy(),
203 security_requirements=dependant.security_requirements.copy(),
204 use_cache=dependant.use_cache,
205 path=dependant.path,
206 )
207 for sub_dependant in dependant.dependencies:
208 if skip_repeats and sub_dependant.cache_key in visited:
209 continue
210 flat_sub = get_flat_dependant(
211 sub_dependant, skip_repeats=skip_repeats, visited=visited
212 )
213 flat_dependant.path_params.extend(flat_sub.path_params)
214 flat_dependant.query_params.extend(flat_sub.query_params)
215 flat_dependant.header_params.extend(flat_sub.header_params)
216 flat_dependant.cookie_params.extend(flat_sub.cookie_params)
217 flat_dependant.body_params.extend(flat_sub.body_params)
218 flat_dependant.security_requirements.extend(flat_sub.security_requirements)
219 return flat_dependant

277def get_dependant(
278 *,
279 path: str,
280 call: Callable[..., Any],
281 name: Optional[str] = None,
282 security_scopes: Optional[List[str]] = None,
283 use_cache: bool = True,
284) -> Dependant:
285 path_param_names = get_path_param_names(path)
286 endpoint_signature = get_typed_signature(call)
287 signature_params = endpoint_signature.parameters
288 dependant = Dependant(
289 call=call,
290 name=name,
291 path=path,
292 security_scopes=security_scopes,
293 use_cache=use_cache,
294 )
295 for param_name, param in signature_params.items():
296 is_path_param = param_name in path_param_names
297 param_details = analyze_param(
298 param_name=param_name,
299 annotation=param.annotation,
300 value=param.default,
301 is_path_param=is_path_param,
302 )
303 if param_details.depends is not None:
304 sub_dependant = get_param_sub_dependant(
305 param_name=param_name,
306 depends=param_details.depends,
307 path=path,
308 security_scopes=security_scopes,
309 )
310 dependant.dependencies.append(sub_dependant)
311 continue
312 if add_non_field_param_to_dependency(
313 param_name=param_name,
314 type_annotation=param_details.type_annotation,
315 dependant=dependant,
316 ):
317 assert param_details.field is None, (
318 f"Cannot specify multiple FastAPI annotations for {param_name!r}"
319 )
320 continue
321 assert param_details.field is not None
322 if isinstance(
323 param_details.field.field_info, (params.Body, temp_pydantic_v1_params.Body)
324 ):
325 dependant.body_params.append(param_details.field)
326 else:
327 add_param_to_fields(field=param_details.field, dependant=dependant)

362def analyze_param(
363 *,
364 param_name: str,
365 annotation: Any,
366 value: Any,
367 is_path_param: bool,
368) -> ParamDetails:
369 field_info = None
370 depends = None
371 type_annotation: Any = Any
372 use_annotation: Any = Any
373 if annotation is not inspect.Signature.empty:
374 use_annotation = annotation
375 type_annotation = annotation
376 # Extract Annotated info
377 if get_origin(use_annotation) is Annotated:
378 annotated_args = get_args(annotation)
379 type_annotation = annotated_args[0]
380 fastapi_annotations = [
381 arg
382 for arg in annotated_args[1:]
383 if isinstance(arg, (FieldInfo, v1.FieldInfo, params.Depends))
384 ]
385 fastapi_specific_annotations = [
386 arg
387 for arg in fastapi_annotations
388 if isinstance(
389 arg,
390 (
391 params.Param,
392 temp_pydantic_v1_params.Param,
393 params.Body,
394 temp_pydantic_v1_params.Body,
395 params.Depends,
396 ),
397 )
398 ]
399 if fastapi_specific_annotations:
400 fastapi_annotation: Union[FieldInfo, v1.FieldInfo, params.Depends, None] = (
401 fastapi_specific_annotations[-1]
402 )
403 else:
404 fastapi_annotation = None
405 # Set default for Annotated FieldInfo
406 if isinstance(fastapi_annotation, (FieldInfo, v1.FieldInfo)):
407 # Copy `field_info` because we mutate `field_info.default` below.
408 field_info = copy_field_info(
409 field_info=fastapi_annotation, annotation=use_annotation
410 )
411 assert field_info.default in {
412 Undefined,
413 v1.Undefined,
414 } or field_info.default in {RequiredParam, v1.RequiredParam}, (
415 f"`{field_info.__class__.__name__}` default value cannot be set in"
416 f" `Annotated` for {param_name!r}. Set the default value with `=` instead."
417 )
418 if value is not inspect.Signature.empty:
419 assert not is_path_param, "Path parameters cannot have default values"
420 field_info.default = value
421 else:
422 field_info.default = RequiredParam
423 # Get Annotated Depends
424 elif isinstance(fastapi_annotation, params.Depends):
425 depends = fastapi_annotation
426 # Get Depends from default value
427 if isinstance(value, params.Depends):
428 assert depends is None, (
429 "Cannot specify `Depends` in `Annotated` and default value"
430 f" together for {param_name!r}"
431 )
432 assert field_info is None, (
433 "Cannot specify a FastAPI annotation in `Annotated` and `Depends` as a"
434 f" default value together for {param_name!r}"
435 )
436 depends = value
437 # Get FieldInfo from default value
438 elif isinstance(value, (FieldInfo, v1.FieldInfo)):
439 assert field_info is None, (
440 "Cannot specify FastAPI annotations in `Annotated` and default value"
441 f" together for {param_name!r}"
442 )
443 field_info = value
444 if PYDANTIC_V2:
445 if isinstance(field_info, FieldInfo):
446 field_info.annotation = type_annotation
447
448 # Get Depends from type annotation
449 if depends is not None and depends.dependency is None:
450 # Copy `depends` before mutating it
451 depends = copy(depends)
452 depends.dependency = type_annotation
453
454 # Handle non-param type annotations like Request
455 if lenient_issubclass(
456 type_annotation,
457 (
458 Request,
459 WebSocket,
460 HTTPConnection,
461 Response,
462 StarletteBackgroundTasks,
463 SecurityScopes,
464 ),
465 ):
466 assert depends is None, f"Cannot specify `Depends` for type {type_annotation!r}"
467 assert field_info is None, (
468 f"Cannot specify FastAPI annotation for type {type_annotation!r}"
469 )
470 # Handle default assignations, neither field_info nor depends was not found in Annotated nor default value
471 elif field_info is None and depends is None:
472 default_value = value if value is not inspect.Signature.empty else RequiredParam
473 if is_path_param:
474 # We might check here that `default_value is RequiredParam`, but the fact is that the same
475 # parameter might sometimes be a path parameter and sometimes not. See
476 # `tests/test_infer_param_optionality.py` for an example.
477 field_info = params.Path(annotation=use_annotation)
478 elif is_uploadfile_or_nonable_uploadfile_annotation(
479 type_annotation
480 ) or is_uploadfile_sequence_annotation(type_annotation):
481 field_info = params.File(annotation=use_annotation, default=default_value)
482 elif not field_annotation_is_scalar(annotation=type_annotation):
483 if annotation_is_pydantic_v1(use_annotation):
484 field_info = temp_pydantic_v1_params.Body(
485 annotation=use_annotation, default=default_value
486 )
487 else:
488 field_info = params.Body(
489 annotation=use_annotation, default=default_value
490 )
491 else:
492 field_info = params.Query(annotation=use_annotation, default=default_value)
493
494 field = None
495 # It's a field_info, not a dependency
496 if field_info is not None:
497 # Handle field_info.in_
498 if is_path_param:
499 assert isinstance(
500 field_info, (params.Path, temp_pydantic_v1_params.Path)
501 ), (
502 f"Cannot use `{field_info.__class__.__name__}` for path param"
503 f" {param_name!r}"
504 )
505 elif (
506 isinstance(field_info, (params.Param, temp_pydantic_v1_params.Param))
507 and getattr(field_info, "in_", None) is None
508 ):
509 field_info.in_ = params.ParamTypes.query
510 use_annotation_from_field_info = get_annotation_from_field_info(
511 use_annotation,
512 field_info,
513 param_name,
514 )
515 if isinstance(field_info, (params.Form, temp_pydantic_v1_params.Form)):
516 ensure_multipart_is_installed()
517 if not field_info.alias and getattr(field_info, "convert_underscores", None):
518 alias = param_name.replace("_", "-")
519 else:
520 alias = field_info.alias or param_name
521 field_info.alias = alias
522 field = create_model_field(
523 name=param_name,
524 type_=use_annotation_from_field_info,
525 default=field_info.default,
526 alias=alias,
527 required=field_info.default in (RequiredParam, v1.RequiredParam, Undefined),
528 field_info=field_info,
529 )
530 if is_path_param:
531 assert is_scalar_field(field=field), (
532 "Path params must be of one of the supported types"
533 )
534 elif isinstance(field_info, (params.Query, temp_pydantic_v1_params.Query)):
535 assert (
536 is_scalar_field(field)
537 or is_scalar_sequence_field(field)
538 or (
539 _is_model_class(field.type_)
540 # For Pydantic v1
541 and getattr(field, "shape", 1) == 1
542 )
543 )
544
545 return ParamDetails(type_annotation=type_annotation, depends=depends, field=field)

548def add_param_to_fields(*, field: ModelField, dependant: Dependant) -> None:
549 field_info = field.field_info
550 field_info_in = getattr(field_info, "in_", None)
551 if field_info_in == params.ParamTypes.path:
552 dependant.path_params.append(field)

561 dependant.cookie_params.append(field)

606async def solve_dependencies(
607 *,
608 request: Union[Request, WebSocket],
609 dependant: Dependant,
610 body: Optional[Union[Dict[str, Any], FormData]] = None,
611 background_tasks: Optional[StarletteBackgroundTasks] = None,
612 response: Optional[Response] = None,
613 dependency_overrides_provider: Optional[Any] = None,
614 dependency_cache: Optional[Dict[Tuple[Callable[..., Any], Tuple[str]], Any]] = None,
615 async_exit_stack: AsyncExitStack,
616 embed_body_fields: bool,
617) -> SolvedDependency:
618 values: Dict[str, Any] = {}
619 errors: List[Any] = []
620 if response is None:
621 response = Response()
622 del response.headers["content-length"]
623 response.status_code = None # type: ignore
624 if dependency_cache is None:
625 dependency_cache = {}
626 sub_dependant: Dependant
627 for sub_dependant in dependant.dependencies:
628 sub_dependant.call = cast(Callable[..., Any], sub_dependant.call)
629 sub_dependant.cache_key = cast(
630 Tuple[Callable[..., Any], Tuple[str]], sub_dependant.cache_key
631 )
632 call = sub_dependant.call
633 use_sub_dependant = sub_dependant
634 if (
635 dependency_overrides_provider
636 and dependency_overrides_provider.dependency_overrides
637 ):
638 original_call = sub_dependant.call
639 call = getattr(
640 dependency_overrides_provider, "dependency_overrides", {}
641 ).get(original_call, original_call)
642 use_path: str = sub_dependant.path # type: ignore
643 use_sub_dependant = get_dependant(
644 path=use_path,
645 call=call,
646 name=sub_dependant.name,
647 security_scopes=sub_dependant.security_scopes,
648 )
649
650 solved_result = await solve_dependencies(
651 request=request,
652 dependant=use_sub_dependant,
653 body=body,
654 background_tasks=background_tasks,
655 response=response,
656 dependency_overrides_provider=dependency_overrides_provider,
657 dependency_cache=dependency_cache,
658 async_exit_stack=async_exit_stack,
659 embed_body_fields=embed_body_fields,
660 )
661 background_tasks = solved_result.background_tasks
662 if solved_result.errors:
663 errors.extend(solved_result.errors)
664 continue
665 if sub_dependant.use_cache and sub_dependant.cache_key in dependency_cache:
666 solved = dependency_cache[sub_dependant.cache_key]
667 elif is_gen_callable(call) or is_async_gen_callable(call):
668 solved = await solve_generator(
669 call=call, stack=async_exit_stack, sub_values=solved_result.values
670 )
671 elif is_coroutine_callable(call):
672 solved = await call(**solved_result.values)
673 else:
674 solved = await run_in_threadpool(call, **solved_result.values)
675 if sub_dependant.name is not None:
676 values[sub_dependant.name] = solved
677 if sub_dependant.cache_key not in dependency_cache:
678 dependency_cache[sub_dependant.cache_key] = solved
679 path_values, path_errors = request_params_to_args(
680 dependant.path_params, request.path_params
681 )
682 query_values, query_errors = request_params_to_args(
683 dependant.query_params, request.query_params
684 )
685 header_values, header_errors = request_params_to_args(
686 dependant.header_params, request.headers
687 )
688 cookie_values, cookie_errors = request_params_to_args(
689 dependant.cookie_params, request.cookies
690 )
691 values.update(path_values)
692 values.update(query_values)
693 values.update(header_values)
694 values.update(cookie_values)
695 errors += path_errors + query_errors + header_errors + cookie_errors
696 if dependant.body_params:
697 (
698 body_values,
699 body_errors,
700 ) = await request_body_to_args( # body_params checked above
701 body_fields=dependant.body_params,
702 received_body=body,
703 embed_body_fields=embed_body_fields,
704 )
705 values.update(body_values)
706 errors.extend(body_errors)

823 params_to_process[key] = value

844 field=field, value=value, values=values, loc=loc

848 else:
849 values[field.name] = v_

872def _should_embed_body_fields(fields: List[ModelField]) -> bool:
873 if not fields:
874 return False
875 # More than one dependency could have the same field, it would show up as multiple
876 # fields but it's the same one, so count them by name
877 body_param_names_set = {field.name for field in fields}
878 # A top level field has to be a single field, not multiple
879 if len(body_param_names_set) > 1:
880 return True
881 first_field = fields[0]
882 # If it explicitly specifies it is embedded, it has to be embedded
883 if getattr(first_field.field_info, "embed", None):
884 return True
885 # If it's a Form (or File) field, it has to be a BaseModel (or a union of BaseModels) to be top level
886 # otherwise it has to be embedded, so that the key value pair can be extracted
887 if (
888 isinstance(first_field.field_info, (params.Form, temp_pydantic_v1_params.Form))
889 and not _is_model_class(first_field.type_)
890 and not is_union_of_base_models(first_field.type_)
891 ):
892 return True
893 return False

898 received_body: FormData,
