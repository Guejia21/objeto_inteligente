104def create_ssl_context(
105 certfile: str | os.PathLike[str],
106 keyfile: str | os.PathLike[str] | None,
107 password: str | None,
108 ssl_version: int,
109 cert_reqs: int,
110 ca_certs: str | os.PathLike[str] | None,
111 ciphers: str | None,
112) -> ssl.SSLContext:
113 ctx = ssl.SSLContext(ssl_version)
114 get_password = (lambda: password) if password else None
115 ctx.load_cert_chain(certfile, keyfile, get_password)
116 ctx.verify_mode = ssl.VerifyMode(cert_reqs)
117 if ca_certs:
118 ctx.load_verify_locations(ca_certs)
119 if ciphers:
120 ctx.set_ciphers(ciphers)
121 return ctx

133def resolve_reload_patterns(patterns_list: list[str], directories_list: list[str]) -> tuple[list[str], list[Path]]:
134 directories: list[Path] = list(set(map(Path, directories_list.copy())))
135 patterns: list[str] = patterns_list.copy()
136
137 current_working_directory = Path.cwd()
138 for pattern in patterns_list:
139 # Special case for the .* pattern, otherwise this would only match
140 # hidden directories which is probably undesired
141 if pattern == ".*":
142 continue # pragma: py-not-linux
143 patterns.append(pattern)
144 if is_dir(Path(pattern)):
145 directories.append(Path(pattern))
146 else:
147 for match in current_working_directory.glob(pattern):
148 if is_dir(match):
149 directories.append(match)
150
151 directories = list(set(directories))
152 directories = list(map(Path, directories))
153 directories = list(map(lambda x: x.resolve(), directories))
154 directories = list({reload_path for reload_path in directories if is_dir(reload_path)})
155
156 children = []
157 for j in range(len(directories)):
158 for k in range(j + 1, len(directories)): # pragma: full coverage
159 if directories[j] in directories[k].parents:
160 children.append(directories[k])
161 elif directories[k] in directories[j].parents:
162 children.append(directories[j])
163
164 directories = list(set(directories).difference(set(children)))
165
166 return list(set(patterns)), directories
