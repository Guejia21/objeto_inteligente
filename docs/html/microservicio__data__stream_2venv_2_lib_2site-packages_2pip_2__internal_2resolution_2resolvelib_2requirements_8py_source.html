61 def format_for_error(self) -> str:
62 # Convert comma-separated specifiers into "A, B, ..., F and G"
63 # This makes the specifier a bit more "human readable", without
64 # risking a change in meaning. (Hopefully! Not all edge cases have
65 # been checked)
66 parts = [s.strip() for s in str(self).split(",")]
67 if len(parts) == 0:
68 return ""
69 elif len(parts) == 1:
70 return parts[0]
71
72 return ", ".join(parts[:-1]) + " and " + parts[-1]
