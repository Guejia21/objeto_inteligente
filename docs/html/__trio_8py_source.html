452 if not isinstance(msglen, int) or msglen < 0:
453 raise ValueError("msglen must be a non-negative integer")
454 if not isinstance(maxfds, int) or maxfds < 1:
455 raise ValueError("maxfds must be a positive integer")

459 with self._receive_guard:

463 msglen, socket.CMSG_LEN(maxfds * fds.itemsize)
464 )
465 except BaseException as exc:
466 self._convert_socket_error(exc)
467 else:
468 if not message and not ancdata:
469 raise EndOfStream
470

474 if cmsg_level != socket.SOL_SOCKET or cmsg_type != socket.SCM_RIGHTS:
475 raise RuntimeError(

477 f"cmsg_level = {cmsg_level}, cmsg_type = {cmsg_type}"

480 fds.frombytes(cmsg_data[: len(cmsg_data) - (len(cmsg_data) % fds.itemsize)])
