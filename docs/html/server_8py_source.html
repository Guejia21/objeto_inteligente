234 # Update the default headers, once per second.
235 if counter % 10 == 0:

237 current_date = formatdate(current_time, usegmt=True).encode()

246 # Callback to `callback_notify` once every `timeout_notify` seconds.
247 if self.config.callback_notify is not None:

314 def capture_signals(self) -> Generator[None, None, None]:
315 # Signals can only be listened to from the main thread.
316 if threading.current_thread() is not threading.main_thread():
317 yield
318 return
319 # always use signal.signal, even if loop.add_signal_handler is available
320 # this allows to restore previous signal handlers later on
321 original_handlers = {sig: signal.signal(sig, self.handle_exit) for sig in HANDLED_SIGNALS}
322 try:
323 yield
324 finally:
325 for sig, handler in original_handlers.items():
326 signal.signal(sig, handler)
327 # If we did gracefully shut down due to a signal, try to
328 # trigger the expected behaviour now; multiple signals would be
329 # done LIFO, see https://stackoverflow.com/questions/48434964
330 for captured_signal in reversed(self._captured_signals):
331 signal.raise_signal(captured_signal)
