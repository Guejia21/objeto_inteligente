436def get_fields_from_routes(
437 routes: Sequence[BaseRoute],
438) -> List[ModelField]:
439 body_fields_from_routes: List[ModelField] = []
440 responses_from_routes: List[ModelField] = []
441 request_fields_from_routes: List[ModelField] = []
442 callback_flat_models: List[ModelField] = []
443 for route in routes:
444 if getattr(route, "include_in_schema", None) and isinstance(
445 route, routing.APIRoute
446 ):
447 if route.body_field:
448 assert _is_model_field(route.body_field), (
449 "A request body must be a Pydantic Field"
450 )

452 if route.response_field:

456 if route.callbacks:
457 callback_flat_models.extend(get_fields_from_routes(route.callbacks))
458 params = get_flat_params(route.dependant)
459 request_fields_from_routes.extend(params)
460
461 flat_models = callback_flat_models + list(
462 body_fields_from_routes + responses_from_routes + request_fields_from_routes
