41 """Initializes the application.
42
43 Parameters:
44 debug: Boolean indicating if debug tracebacks should be returned on errors.
45 routes: A list of routes to serve incoming HTTP and WebSocket requests.
46 middleware: A list of middleware to run for every request. A starlette
47 application will always automatically include two middleware classes.
48 `ServerErrorMiddleware` is added as the very outermost middleware, to handle
49 any uncaught errors occurring anywhere in the entire stack.
50 `ExceptionMiddleware` is added as the very innermost middleware, to deal
51 with handled exception cases occurring in the routing or endpoints.
52 exception_handlers: A mapping of either integer status codes,
53 or exception class types onto callables which handle the exceptions.
54 Exception handler callables should be of the form
55 `handler(request, exc) -> response` and may be either standard functions, or
56 async functions.
57 on_startup: A list of callables to run on application startup.
58 Startup handler callables do not take any arguments, and may be either
59 standard functions, or async functions.
60 on_shutdown: A list of callables to run on application shutdown.
61 Shutdown handler callables do not take any arguments, and may be either
62 standard functions, or async functions.
63 lifespan: A lifespan context function, which can be used to perform
64 startup and shutdown tasks. This is a newer style that replaces the
65 `on_startup` and `on_shutdown` handlers. Use one or the other, not both.
66 """

80 def build_middleware_stack(self) -> ASGIApp:
81 debug = self.debug
82 error_handler = None
83 exception_handlers: dict[Any, ExceptionHandler] = {}
84
85 for key, value in self.exception_handlers.items():
86 if key in (500, Exception):
87 error_handler = value
88 else:
89 exception_handlers[key] = value

91 middleware = (
92 [Middleware(ServerErrorMiddleware, handler=error_handler, debug=debug)]
93 + self.user_middleware
94 + [Middleware(ExceptionMiddleware, handlers=exception_handlers, debug=debug)]

98 for cls, args, kwargs in reversed(middleware):
99 app = cls(app, *args, **kwargs)
