357 @no_type_check
358 def __setattr__(self, name, value): # noqa: C901 (ignore complexity)
359 if name in self.__private_attributes__ or name in DUNDER_ATTRIBUTES:
360 return object_setattr(self, name, value)
361
362 if self.__config__.extra is not Extra.allow and name not in self.__fields__:
363 raise ValueError(f'"{self.__class__.__name__}" object has no field "{name}"')
364 elif not self.__config__.allow_mutation or self.__config__.frozen:
365 raise TypeError(f'"{self.__class__.__name__}" is immutable and does not support item assignment')
366 elif name in self.__fields__ and self.__fields__[name].final:
367 raise TypeError(
368 f'"{self.__class__.__name__}" object "{name}" field is final and does not support reassignment'
369 )
370 elif self.__config__.validate_assignment:
371 new_values = {**self.__dict__, name: value}
372
373 for validator in self.__pre_root_validators__:
374 try:
375 new_values = validator(self.__class__, new_values)
376 except (ValueError, TypeError, AssertionError) as exc:
377 raise ValidationError([ErrorWrapper(exc, loc=ROOT_KEY)], self.__class__)
378
379 known_field = self.__fields__.get(name, None)
380 if known_field:
381 # We want to
382 # - make sure validators are called without the current value for this field inside `values`
383 # - keep other values (e.g. submodels) untouched (using `BaseModel.dict()` will change them into dicts)
384 # - keep the order of the fields
385 if not known_field.field_info.allow_mutation:
386 raise TypeError(f'"{known_field.name}" has allow_mutation set to False and cannot be assigned')
387 dict_without_original_value = {k: v for k, v in self.__dict__.items() if k != name}
388 value, error_ = known_field.validate(value, dict_without_original_value, loc=name, cls=self.__class__)
389 if error_:
390 raise ValidationError([error_], self.__class__)
391 else:
392 new_values[name] = value
393
394 errors = []
395 for skip_on_failure, validator in self.__post_root_validators__:
396 if skip_on_failure and errors:
397 continue
398 try:
399 new_values = validator(self.__class__, new_values)
400 except (ValueError, TypeError, AssertionError) as exc:
401 errors.append(ErrorWrapper(exc, loc=ROOT_KEY))
402 if errors:
403 raise ValidationError(errors, self.__class__)
404
405 # update the whole __dict__ as other values than just `value`
406 # may be changed (e.g. with `root_validator`)
407 object_setattr(self, '__dict__', new_values)
408 else:
409 self.__dict__[name] = value
410
411 self.__fields_set__.add(name)

591 @classmethod
592 def construct(cls: Type['Model'], _fields_set: Optional['SetStr'] = None, **values: Any) -> 'Model':
593 """
594 Creates a new model setting __dict__ and __fields_set__ from trusted or pre-validated data.
595 Default values are respected, but no other validation is performed.
596 Behaves as if `Config.extra = 'allow'` was set since it adds all passed values
597 """
598 m = cls.__new__(cls)
599 fields_values: Dict[str, Any] = {}
600 for name, field in cls.__fields__.items():
601 if field.alt_alias and field.alias in values:
602 fields_values[name] = values[field.alias]
603 elif name in values:
604 fields_values[name] = values[name]
605 elif not field.required:
606 fields_values[name] = field.get_default()
607 fields_values.update(values)
608 object_setattr(m, '__dict__', fields_values)
609 if _fields_set is None:
610 _fields_set = set(values.keys())
611 object_setattr(m, '__fields_set__', _fields_set)
612 m._init_private_attributes()
613 return m
