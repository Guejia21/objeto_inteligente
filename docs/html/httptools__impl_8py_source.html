450 async def send(self, message: ASGISendEvent) -> None:
451 message_type = message["type"]
452
453 if self.flow.write_paused and not self.disconnected:
454 await self.flow.drain() # pragma: full coverage
455
456 if self.disconnected:
457 return # pragma: full coverage
458
459 if not self.response_started:
460 # Sending response status line and headers
461 if message_type != "http.response.start":
462 msg = "Expected ASGI message 'http.response.start', but got '%s'."
463 raise RuntimeError(msg % message_type)
464 message = cast("HTTPResponseStartEvent", message)
465
466 self.response_started = True
467 self.waiting_for_100_continue = False
468
469 status_code = message["status"]
470 headers = self.default_headers + list(message.get("headers", []))
471
472 if CLOSE_HEADER in self.scope["headers"] and CLOSE_HEADER not in headers:
473 headers = headers + [CLOSE_HEADER]
474
475 if self.access_log:
476 self.access_logger.info(
477 '%s - "%s %s HTTP/%s" %d',
478 get_client_addr(self.scope),
479 self.scope["method"],
480 get_path_with_query_string(self.scope),
481 self.scope["http_version"],
482 status_code,
483 )
484
485 # Write response status line and headers
486 content = [STATUS_LINE[status_code]]
487
488 for name, value in headers:
489 if HEADER_RE.search(name):
490 raise RuntimeError("Invalid HTTP header name.") # pragma: full coverage
491 if HEADER_VALUE_RE.search(value):
492 raise RuntimeError("Invalid HTTP header value.")
493
494 name = name.lower()
495 if name == b"content-length" and self.chunked_encoding is None:
496 self.expected_content_length = int(value.decode())
497 self.chunked_encoding = False
498 elif name == b"transfer-encoding" and value.lower() == b"chunked":
499 self.expected_content_length = 0
500 self.chunked_encoding = True
501 elif name == b"connection" and value.lower() == b"close":
502 self.keep_alive = False
503 content.extend([name, b": ", value, b"\r\n"])
504
505 if self.chunked_encoding is None and self.scope["method"] != "HEAD" and status_code not in (204, 304):
506 # Neither content-length nor transfer-encoding specified
507 self.chunked_encoding = True
508 content.append(b"transfer-encoding: chunked\r\n")
509
510 content.append(b"\r\n")
511 self.transport.write(b"".join(content))
512
513 elif not self.response_complete:
514 # Sending response body
515 if message_type != "http.response.body":
516 msg = "Expected ASGI message 'http.response.body', but got '%s'."
517 raise RuntimeError(msg % message_type)
518
519 body = cast(bytes, message.get("body", b""))
520 more_body = message.get("more_body", False)
521
522 # Write response body
523 if self.scope["method"] == "HEAD":
524 self.expected_content_length = 0
525 elif self.chunked_encoding:
526 if body:
527 content = [b"%x\r\n" % len(body), body, b"\r\n"]
528 else:
529 content = []
530 if not more_body:
531 content.append(b"0\r\n\r\n")
532 self.transport.write(b"".join(content))
533 else:
534 num_bytes = len(body)
535 if num_bytes > self.expected_content_length:
536 raise RuntimeError("Response content longer than Content-Length")
537 else:
538 self.expected_content_length -= num_bytes
539 self.transport.write(body)
540
541 # Handle response completion
542 if not more_body:
543 if self.expected_content_length != 0:
544 raise RuntimeError("Response content shorter than Content-Length")
545 self.response_complete = True
546 self.message_event.set()
547 if not self.keep_alive:
548 self.transport.close()
549 self.on_response()
550
551 else:
552 # Response already sent
553 msg = "Unexpected ASGI message '%s' sent, after response already completed."
554 raise RuntimeError(msg % message_type)
