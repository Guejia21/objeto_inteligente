63 def _load_metadata(self):
64 """
65 * @brief Carga configuración de datastreams desde archivo metadata.json
66 * @details
67 * Estructura esperada de metadata.json:
68 * {
69 * "object": {
70 * "id": "ESP32_Sala",
71 * "title": "Sensor de Sala",
72 * "ip_object": "192.168.1.100",
73 * "thingsboard_token": "token_aqui"

75 * "datastreams": [
76 * {
77 * "datastream_id": "temperatura",
78 * "datastream_format": "float",
79 * "datastream_type": "sensor"
80 * },
81 * ...
82 * ]
83 * }
84 *
85 * @return void
86 * @throws Exception Si archivo no existe o JSON inválido (capturado internamente)
87 * @see Config.PATH_METADATA ubicación del archivo
88 * @see self.datastreams almacena lista de datastreams
89 """
90 """Carga metadata del objeto desde JSON"""
91 try:
92 with open(f"{Config.PATH_METADATA}metadata.json", 'r') as f:
93 metadata = json.load(f)
94 Config.OSID = metadata['object']['id']
95 Config.TITLE = metadata['object'].get('title', 'Unknown')
96 Config.OBJECT_IP = metadata['object'].get('ip_object', 'Unknown')
97 Config.THINGSBOARD_ACCESS_TOKEN = metadata['object'].get('thingsboard_token', '')
98 self.datastreams = metadata.get('datastreams', [])
99 print(f"Metadata cargada: OSID={Config.OSID}, Datastreams={len(self.datastreams)}")
100 except Exception as e:
101 print(f"Error cargando metadata: {e}")
102 self.datastreams = []

105 """
106 * @brief Verifica si un datastream está registrado en configuración

122 """

167 variable_estado,

169 value

227 def set_datastream(self, osid, id_datastream, comando):
228 """
229 Establece el valor de un datastream (actuador)

231 Args:
232 osid: ID del objeto

254 print(f"SetDatastream: {id_datastream} -> {comando}")

256 success = False

284 Config.CODES['errorDatastream']
