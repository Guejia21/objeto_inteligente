99 async def _process_telemetry_message(self, message: str):
100 """Procesa un mensaje de telemetría y evalúa todos los ECAs activos"""
101 try:
102 #logger.debug(f"Procesando mensaje de telemetría: {message}")

111 continue

114 expected_action = "on" if evaluator.action_value else "off"

120 # La condición ya no se cumple, revertir acción si fue ejecutada

123 await evaluator.execute_inverse_action()

125 except json.JSONDecodeError as e:
